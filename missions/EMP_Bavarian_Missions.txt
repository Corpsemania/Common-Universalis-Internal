

emp_bav_prestige_branch = {
	slot = 2
	generic = no
	ai = yes
	potential_on_load = {
		has_dlc = "Emperor"
	}
	potential = {
		has_dlc = "Emperor"
		OR = {
			tag = BAV
			tag = ING
			tag = LBV
			tag = UBV
		}
		NOT = { map_setup = map_setup_random }
	}
	has_country_shield = yes
	
	emp_bav_cultural_center = {
		icon = develop_our_capital
		required_missions = { }
		provinces_to_highlight = {
			owned_by = root
			is_capital = yes
			OR = {
				NOT = { development = 25 }
				NOT = { num_of_buildings_in_province = 5 }
			}
		}
		trigger = {
			capital_scope = {
				development = 25
				num_of_buildings_in_province = 5
			}
		}
		effect = {
			capital_scope = {
				if = {
					limit = {
						province_has_center_of_trade_of_level = 1
						NOT = { province_has_center_of_trade_of_level = 3 }
					}
					add_center_of_trade_level = 1
				}
				else_if = {
					limit = {
						NOT = { province_has_center_of_trade_of_level = 1 }
					}
					center_of_trade = 1
				}
				else = {
					add_base_tax = 1
					add_base_production = 1
					add_base_manpower = 1
				}
			}
		}
	}
	emp_bav_splendour = {
		icon = regal_splendour
		required_missions = { emp_bav_cultural_center }
		trigger = {
			if = {
				limit = { has_dlc = "Common Sense" }
				government_rank = 2
			}
			else = { total_development = 300 }
			is_subject = no
			if = {
				limit = {
					has_dlc = "Mandate of Heaven"
					has_had_golden_age = no
				}
				in_golden_age = yes
			}
			else = {
				prestige = 80
			}
			legitimacy_equivalent = 95
		}
		effect = {
			add_prestige = 20

			add_country_modifier = {
				name = bav_splendorous_bavaria
				duration = 7300
			}

			if = {
				limit = {
					NOT = { has_dlc = "Common Sense" }
					NOT = { government_rank = 2 }
					OR = {
						is_elector = yes
						is_part_of_hre = no
					}
				}
				set_government_rank = 2
			}
		}
	}
	emp_bav_oktoberfest_mission = {
		icon = establish_oktoberfest
		required_missions = { emp_bav_splendour }
		provinces_to_highlight = {
			OR = {
				AND = {
					owned_by = root
					OR = {	
						trade_goods = grain
						trade_goods = livestock
						trade_goods = wine
					}
					NOT = { has_building = farm_estate }
				}
				AND = {
					ROOT = {
						NOT = {
							num_of_owned_provinces_with = {
								value = 5
								OR = {	
									trade_goods = grain
									trade_goods = livestock
									trade_goods = wine
								}
							}
						}
					}
					OR = {	
						trade_goods = grain
						trade_goods = livestock
						trade_goods = wine
					}
					OR = {
						owner = { is_subject_of = ROOT }
						any_neighbor_province = {
							owned_by = ROOT
						}
					}
				}
			}
		}
		trigger = {
			num_of_owned_provinces_with = {
				value = 5 #eligible Bavarian starting provs (of all three duchies)
				has_building = farm_estate
			}
			trade_goods_produced_amount = {
				trade_goods = grain
				amount = 25
			}
		}
		effect = {
			capital_scope = {
				add_province_modifier = {
					name = bav_established_oktoberfest
					duration = -1
				}
				if = {
					limit = {
						NOT = { devastation = 1 }
						NOT = { prosperity = 100 }
						ROOT = { has_dlc = "Mandate of Heaven" }
					}
					add_prosperity = 50
				}
			}
		}
	}
}

					}
