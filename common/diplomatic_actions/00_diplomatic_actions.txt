# INSTRUCTIONS:
#
# condition				A diplomatic action can have any number of condition blocks, each with its own
#						tooltip, potential and allow section
#
# 	tooltip					Sets a custom text string similar to the hardcoded limits
# 							If no tooltip is scripted, the tooltip for the actual trigger will be shown
#							Note that the custom tooltip is only shown if the allow trigger is NOT met
#
# 	potential				Determines if the trigger is applicable or not
# 	allow					Determines if the action is valid or not
#
# effect				A diplomatic action can only have one effect block

# ROOT					actor
# FROM					target

declarewar = {
    condition = {
		tooltip = OBSERVER_DIP
		potential = {
                OR = {
				     has_country_flag = cu_observer_country
					 FROM = { has_country_flag = cu_observer_country }
				}
		}
		allow = {
		        NOT = { has_global_flag = cu_multiplayer_mode }	
		}
	}
    condition = {
		tooltip = DECLAREWAR_HRE
		potential = {
                        #FROM = { has_regency = no }
		}
		allow = {
		    OR = {
			        ai = no
				    dip = 2
				    mil = 2
					ruler_has_personality = bold_fighter_personality
					is_part_of_hre = no
                    FROM = { is_part_of_hre = no }
			}
			OR = { 
			                    
			                    AND = {
										 NOT = {
								                any_country = {
												     is_emperor = yes
					                                 alliance_with = ROOT
                                                }          		 
										 }
								}
                                is_part_of_hre = no
                                FROM = { is_part_of_hre = no }
                                is_emperor = yes
                                FROM = { is_emperor = yes }
                                is_rival = FROM
                                FROM = {
                                        is_rival = ROOT
                                }
                                AND = {
                                        ai = yes
                                        is_year = 1600
                                        OR = {
                                               tag = BRA
                                               tag = PRU
                                               tag = GER
                                        }
                                }
                                any_country = {
					                    is_emperor = yes
                                        OR = {
                                               is_rival = FROM
                                               FROM = {
                                                        is_rival = PREV
                                               }
                                        }
				                }
                                any_country = {
					                         is_emperor = yes
                                             OR = {
                                                  is_rival = ROOT
                                                  ROOT = {
                                                        is_rival = PREV
                                                  }
                                             }
                                }
                                is_league_enemy = FROM
                                FROM = {
                                        any_owned_province = {
					                             is_core = ROOT
                                        }
				                }
                                has_opinion_modifier = {
			                        modifier = insulted
			                        who = FROM
		                        }
                                FROM = {  # Coalition
                                         coalition_target = ROOT
                                         NOT = { ROOT = { is_in_coalition_war = yes } }
		                                 num_of_coalition_members = 3
                                }
                                is_revolution_target = yes # Revolution
                                FROM = { is_revolution_target = yes } # Revolution
                        }
		}
	}
	condition = {
		tooltip = DECLAREWAR_CELESTIAL
		potential = {
                    is_subject_of_type = tributary_state
					FROM = { is_subject_of_type = tributary_state }
		}
		allow = {
		    NOT = { has_global_flag = celestial_peace_enforced }
		}
	}
	condition = { #Prevent AI from over expanding too much
		tooltip = AI_EXPANSION_LIMIT
		potential = {
                ai = yes
		}
		allow = {
				OR = {
					FROM = { ai = no }
					FROM = {  # Coalition
                                         coalition_target = ROOT
                                         NOT = { ROOT = { is_in_coalition_war = yes } }
		                                 num_of_coalition_members = 3
                    }
					has_opinion_modifier = {
			                        modifier = insulted
			                        who = FROM
		            }
					OR = {
                                               is_rival = FROM
                                               FROM = {
                                                        is_rival = PREV
                                               }
                    }
                    is_revolution_target = yes # Revolution
					NOT = { check_variable = { which = go value = 130 } }	
				}
		}
	}
}
