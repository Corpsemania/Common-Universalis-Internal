					value = 5

					NOT = { province_religion = ROOT }
				}
			}
		}
		modifier = {
			factor = 3
			num_of_owned_provinces_with = {
				value = 5
				region = low_countries_region
				NOT = { province_religion = ROOT }
			}
		}

			factor = 1
			any_owned_province = {
				region = low_countries_region

			NOT = {
				num_of_owned_provinces_with = {

					value = 5

			}
		}

			factor = 3
			num_of_owned_provinces_with = {
				value = 5
