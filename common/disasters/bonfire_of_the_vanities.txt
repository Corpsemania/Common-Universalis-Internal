bonfire_of_the_vanities_disaster = {
	potential = {
		normal_or_historical_nations = yes
		OR = {
			tag = LAN
			tag = TUS
		}
		NOT = {	is_year = 1550 }
		capital = 116
		OR = {
			has_reform = signoria_reform
			AND = {
				government = monarchy
				dynasty = "de' Medici"
			}
		}
		religion = catholic
		NOT = { has_reform = celestial_empire }
		uses_doom = no
		is_free_or_tributary_trigger = yes
		NOT = { has_country_flag = bonfire_of_the_vanities_disaster_flag }
	}
	
	can_start = {
		NOT = { stability = 2 }
		is_at_war = no
		if = {
			limit = {
				government = monarchy
			}
			NOT = { legitimacy = 80 }
		}
		else = {
			NOT = { republican_tradition = 80 }
		}
	}
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			is_year = 1550
			OR = {
				republican_tradition = 90
				legitimacy = 90
			}
			AND = {
				government = monarchy
				NOT = { dynasty = "de' Medici" }
			}
		}
	}
	
	progress = {
		modifier = {
			factor = 5
			NOT = { stability = 1 }
		}
		modifier = {
			factor = 5
			NOT = { stability = 0 }
		}
		modifier = {
			factor = 10
			NOT = { stability = -1 }
		}
		modifier = {
			factor = 10
			government = republic
			NOT = { republican_tradition = 80 }
		}
		modifier = {
			factor = 10
			government = republic
			NOT = { republican_tradition = 50 }
		}
		modifier = {
			factor = 10
			government = republic
			NOT = { republican_tradition = 30 }
		}
		modifier = {
			factor = 10
			government = monarchy
			NOT = { legitimacy = 80 }
		}
		modifier = {
			factor = 10
			government = monarchy
			NOT = { legitimacy = 50 }
		}
		modifier = {
			factor = 10
			government = monarchy
			NOT = { legitimacy = 30 }
		}
	}
	
	can_end = {
		custom_trigger_tooltip = {
			tooltip = had_disaster_for_6_months
			had_country_flag = { flag = bonfire_of_the_vanities_disaster_flag days = 182 }
		}
		OR = {
			custom_trigger_tooltip = {
				tooltip = savonarola_dead_tt
				has_country_flag = savonarola_dead
			}
			government = theocracy
		}
	}
	
	modifier = {
		global_unrest = 5
		stability_cost_modifier = 0.5
	}
	
	on_start = flavor_tus.9
	on_end = flavor_tus.17
	
	on_monthly = {
		events = {
			flavor_tus.15 # people turn against savonarola
			flavor_tus.18 # Savonarola the Prophet
			flavor_tus.19 # Papal condemnation
		}
		random_events = {
			2000 = 0
			50 = flavor_tus.14 # reforming the republic
			50 = flavor_tus.13 # the people follow God
			50 = flavor_tus.12 # people expect revelations
			50 = flavor_tus.11 # medici agents
			50 = flavor_tus.10 # bonfire of the vanities
			50 = flavor_tus.16 #Savanorola Leaves This World
		}
	}
}
