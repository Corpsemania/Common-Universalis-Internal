road_1 = {
	cost = 120
	time = 24
	modifier = {
		local_development_cost = -0.15
        province_trade_power_value = 2
		allowed_num_of_buildings = 2
		local_friendly_movement_speed = 0.15
		local_hostile_movement_speed = 0.1
        supply_limit_modifier = 0.2
		local_state_maintenance_modifier = -0.05
		global_tax_income = -1.8
	}
	on_built = {
	     #custom_tooltip = cu_building_bca_5
	     hidden_effect = {
		     set_variable = { which = cu_var1 value = 5 }
		     cu_add_cultural_acceptance = yes
			 cu_prov_trigger_dist_calc = yes
			 province_event = { id = ideagroups.4894  }
			 
		 }
	     
	}
	
	on_destroyed = {
	     #custom_tooltip = cu_building_dca_n7
	     hidden_effect = {
		     set_variable = { which = cu_var1 value = -7 }
		     cu_add_cultural_acceptance = yes
			 cu_prov_trigger_dist_calc = yes
			 remove_province_modifier = quantity_ideas_5_mod
		 }
	}
	
	on_obsolete = {
         hidden_effect = {
		     set_variable = { which = cu_var1 value = -5 }
		     cu_add_cultural_acceptance = yes
		 }
    }
	ai_will_do = {
		factor = 400
		modifier = {
			factor = 1.5
			any_neighbor_province = {
					OR  = {
					       has_building = road_1
						   has_building = road_2
						   has_building = road_3
					}
			}
		}
		modifier = {
			factor = 3
			is_capital = yes
		}
		modifier = {
			factor = 0.2
			NOT = { development = 10 }
		}
		modifier = {
			factor = 0.5
			NOT = { development = 15 }
		}	
		modifier = {
			factor = 1.5
			development = 20
		}
		modifier = {
			factor = 1.5
			development = 25
		}
		modifier = {
			factor = 1.5
			development = 30
		}
		modifier = {
			factor = 0.25
			owner = { 
			     government = tribal
			}
		}
		modifier = {
			factor = 0.5
			owner = { 
			     is_nomad = yes
			}
		}
		modifier = {
			factor = 2
			OR  = {
					       has_building = fort_15th
						   has_building = fort_16th
						   has_building = fort_17th
						   has_building = fort_18th
			}
		}
		modifier = {
			factor = 0.5
			NOT = { has_owner_culture = yes }
			NOT = { has_owner_accepted_culture = yes }
			NOT = { owner = { culture_group = ROOT } }
		}
	}
}
