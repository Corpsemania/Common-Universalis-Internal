tax_policy_1 = {
	icon = "subsidize_my_love"
	allow_normal_conversion = yes
	#legacy_equivalent = feudal_monarchy
	valid_for_nation_designer = yes
	nation_designer_cost = 0
	valid_for_new_country = yes
	trigger = {
	    
		OR = {
			has_reform = tax_policy_1
			AND = {
				cu_didnt_change_tax_policy = yes  
				ai = no
				adm_tech = 3
				custom_trigger_tooltip = {
					tooltip = has_tax_policy_2
					has_country_flag = tax_policy_2
				}
				adm_power = 50
			}
		}
	}
	modifiers = {
		global_tax_modifier = -0.4
        production_efficiency = -0.4
        development_cost = -0.15
		trade_company_investment_cost = -0.15
		global_unrest = -3
		yearly_absolutism = 0.10
	}
	effect = {
		clr_ruler_flag = chose_tax_policy 
        clr_country_flag = chose_tax_policy_timer 
        set_ruler_flag = chose_tax_policy
        set_country_flag = chose_tax_policy_timer 
		set_country_flag = tax_policy_1
		if = { 
			limit = { has_country_flag = tax_policy_2 }
			add_adm_power = -50
			add_legitimacy = 10
			add_meritocracy = 10
			if = {
				limit = { NOT = { has_country_flag = cu_stability_fixed } }
				hidden_effect = {
				   set_variable = { which = cu_var1 value = 25 }
				}
				cu_change_stability_point_hidden = yes
				custom_tooltip = cu_change_stability_p25
				if = {
					limit = {
						has_estate = estate_nobles
					}
					add_estate_loyalty = {
						estate = estate_nobles
						loyalty = -15
					}
				}
				else_if = {
					limit = {
						has_estate = estate_qinwang
					}
					add_estate_loyalty = {
						estate = estate_qinwang
						loyalty = -15
					}
				}
				if = {
					limit = {
						has_estate = estate_church
					}
					add_estate_loyalty = {
						estate = estate_church
						loyalty = -15
					}
				}
				else_if = {
					limit = {
						has_estate = estate_shizu
					}
					add_estate_loyalty = {
						estate = estate_shizu
						loyalty = -15
					}
				}
				if = {
					limit = {
						has_estate = estate_burghers
					}
					add_estate_loyalty = {
						estate = estate_burghers
						loyalty = 15
					}
				}
			}
		}
		hidden_effect = {
			clr_country_flag = tax_policy_2
