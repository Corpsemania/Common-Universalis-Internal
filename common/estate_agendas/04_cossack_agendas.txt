
#This one or the next one will always be valid
estate_cossacks_hire_advisor = {
	can_select = {
		NOT = {
			employed_advisor = {
				category = MIL
			}
		}
	}
	selection_weight = {
		factor = 1
	}
	task_requirements = {
		employed_advisor = {
			category = MIL
		}
		is_in_deficit = no
	}
    task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_cossacks
			loyalty = 10
		}
    }
    failing_effect = {
        add_estate_loyalty_modifier = {
            estate = estate_cossacks
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}

estate_cossacks_fire_advisor = {
	can_select = {
		employed_advisor = {
			category = MIL
		}
	}
	selection_weight = {
		factor = 1
		modifier = {
			factor = 0.5
			always = yes
		}
	}
	immediate_effect = {
		hidden_effect = {
			set_country_flag = need_to_fire_mil_advisor
		}
	}
	on_invalid = {
		clr_country_flag = need_to_fire_mil_advisor
	}
	task_requirements = {
		custom_trigger_tooltip = {
			tooltip = fire_mil_advisor_tooltip
			has_country_flag = just_fired_mil_advisor
		}
	}
    task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_cossacks
			loyalty = 10
		}
		clr_country_flag = just_fired_mil_advisor
		clr_country_flag = need_to_fire_mil_advisor
	}
	invalid_trigger = {
		NOT = {
			employed_advisor = {
				category = MIL
			}
		}
		NOT = { has_country_flag = just_fired_mil_advisor }
	}
    failing_effect = {
        add_estate_loyalty_modifier = {
            estate = estate_cossacks
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
		clr_country_flag = need_to_fire_mil_advisor
    }
}

#Give us 3 privileges
estate_cossacks_three_privileges = {
	can_select = {
		NOT = {
			num_of_estate_privileges = {
				estate = estate_cossacks
				value = 2
			}
		}
	}
	selection_weight = {
		factor = 1
	}
	pre_effect = {
		set_variable = {
			which = estate_cossacks_three_privileges_var
			value = 0
		}
		if = {
			limit = {
				num_of_estate_privileges = {
					estate = estate_cossacks
					value = 1
				}
			}
			set_variable = {
				which = estate_cossacks_three_privileges_var
				value = 2
			}
		}
		else = {
			set_variable = {
				which = estate_cossacks_three_privileges_var
				value = 1
			}
		}
	}
	task_requirements = {
		if = {
			limit = {
				check_variable = {
					which = estate_cossacks_three_privileges_var
					value = 2
				}
			}
			num_of_estate_privileges = {
				estate = estate_cossacks
				value = 3
			}
		}
		else = {
			num_of_estate_privileges = {
				estate = estate_cossacks
				value = 2
			}
		}
	}
    task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_cossacks
			loyalty = 20
		}
    }
    failing_effect = {
        add_estate_loyalty_modifier = {
            estate = estate_cossacks
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}

#Insult X rival
estate_cossacks_reply_of_the_zaporozhian = {
	can_select = {
		any_rival_country = {
			NOT = { has_estate = estate_cossacks }
			army_balance = {
				who = root
				value = 1.25
			}
			NOT = {
				custom_trigger_tooltip = {
					tooltip = has_been_insulted_by_root
					OR = {
						has_opinion_modifier = {
							who = ROOT
							modifier = insulted
						}
						has_opinion_modifier = {
							who = ROOT
							modifier = scornfully_insulted
						}
					}
				}
			}
		}
		NOT = {
			has_country_modifier = recent_estate_cossacks_reply_of_the_zaporozhian_timer
		}
	}
	selection_weight = {
		factor = 1.5
	}
	pre_effect = {
		random_rival_country = {
			limit = {
				NOT = { has_estate = estate_cossacks }
				army_balance = {
					who = root
					value = 1.25
				}
				NOT = {
					custom_trigger_tooltip = {
						tooltip = has_been_insulted_by_root
						OR = {
							has_opinion_modifier = {
								who = ROOT
								modifier = insulted
							}
							has_opinion_modifier = {
								who = ROOT
								modifier = scornfully_insulted
							}
						}
					}
				}
			}
			save_event_target_as = agenda_country
		}
	}
	immediate_effect = {
		hidden_effect = {
			add_country_modifier = {
				name = recent_estate_cossacks_reply_of_the_zaporozhian_timer
				duration = 9125
				hidden = yes
			}
		}
	}
	provinces_to_highlight = {
		is_capital_of = event_target:agenda_country
	}
	task_requirements = {
		if = {
			limit = {
				event_target:agenda_country = {
					num_of_cities = 1
				}
			}
			event_target:agenda_country = {
				custom_trigger_tooltip = {
					tooltip = has_been_insulted_by_root
					OR = {
						has_opinion_modifier = {
							who = ROOT
							modifier = insulted
						}
						has_opinion_modifier = {
							who = ROOT
							modifier = scornfully_insulted
						}
					}
				}
			}
		}
		else = {
			always = yes
		}
	}
    task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_cossacks
			loyalty = 10
		}
		add_army_tradition = 3
    }
    failing_effect = {
        add_estate_loyalty_modifier = {
            estate = estate_cossacks
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}

#Settle random neighbouring steppe province
estate_cossacks_colonise_X = {
	can_select = {
		OR = {
			num_of_colonists = 1
			has_idea = siberian_frontier
		}
		any_owned_province = {
			any_empty_neighbor_province = {
				ROOT = { has_discovered = prev }
				is_city = no
				is_wasteland = no
				OR = {
					region = ural_region
					region = west_siberia_region
					region = east_siberia_region
				}
			}
		}
	}
	selection_weight = {
		factor = 2.5
		modifier = {
			factor = 2
			num_of_colonists = 2
		}
		modifier = {
			factor = 2
			tag = RUS
		}
	}
	pre_effect = {
		random_owned_province = {
			limit = {
				any_empty_neighbor_province = {
					has_discovered = root
					is_city = no
					is_wasteland = no
					OR = {
						region = ural_region
						region = west_siberia_region
						region = east_siberia_region
					}
				}
			}
			random_empty_neighbor_province = {
				limit = {
					has_discovered = root
					is_city = no
					is_wasteland = no
					OR = {
						region = ural_region
						region = west_siberia_region
						region = east_siberia_region
					}
				}
				save_event_target_as = agenda_province
				choose_random_development_bonus_effect = yes
			}
		}
	}
	on_invalid = {
		event_target:agenda_province = {
			clear_random_development_bonus_flags_effect = yes
		}
	}
	provinces_to_highlight = {
		province_id = event_target:agenda_province
	}
	task_requirements = {
		event_target:agenda_province = {
			owned_by = root
			is_city = yes
		}
	}
	task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_cossacks
			loyalty = 10
		}
		event_target:agenda_province = {
			apply_random_development_bonus_bonus_effect = yes
		}
	}
	fail_if = {
		OR = {
			AND = {
				NOT = { num_of_colonists = 1 }
				NOT = { has_idea = siberian_frontier }
				event_target:agenda_province = {
					NOT = { owned_by = root }

#Recruit more cossacks
estate_cossacks_recruit_more_cossacks = {
	can_select = {
		has_estate_privilege = estate_cossacks_establish_the_cossack_regiments
		OR = {
			NOT = { cossacks_percentage = 0.1 }
			AND = {
				has_estate_privilege = estate_cossacks_expand_the_cossack_regiments
				NOT = { cossacks_percentage = 0.2 }
			}
		}
	}
	selection_weight = {
		factor = 1.5
	}
	pre_effect = {
		set_variable = {
			which = estate_cossacks_recruit_more_cossacks_var
			value = 0
		}
		if = {
			limit = {
				cossacks_percentage = 0.15
			}
			set_variable = {
				which = estate_cossacks_recruit_more_cossacks_var
				value = 4
			}
		}
		else_if = {
			limit = {
				cossacks_percentage = 0.1
			}
			set_variable = {
				which = estate_cossacks_recruit_more_cossacks_var
				value = 3
			}
		}
		else_if = {
			limit = {
				cossacks_percentage = 0.05
			}
			set_variable = {
				which = estate_cossacks_recruit_more_cossacks_var
				value = 2
			}
		}
		else = {
			set_variable = {
				which = estate_cossacks_recruit_more_cossacks_var
				value = 1
			}
		}
	}
	task_requirements = {
		if = {
			limit = {
				check_variable = {
					which = estate_cossacks_recruit_more_cossacks_var
					value = 4
				}
			}
			cossacks_percentage = 0.2
		}
		else_if = {
			limit = {
				check_variable = {
					which = estate_cossacks_recruit_more_cossacks_var
					value = 3
				}
			}
			cossacks_percentage = 0.15
		}
		else_if = {
			limit = {
				check_variable = {
					which = estate_cossacks_recruit_more_cossacks_var
					value = 2
				}
			}
			cossacks_percentage = 0.1
		}
		else = {
			cossacks_percentage = 0.05
		}
	}
	fail_if = {
		NOT = { has_estate_privilege = estate_cossacks_establish_the_cossack_regiments }
	}
    task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_cossacks
			loyalty = 10
		}
		add_army_tradition = 10
    }
    failing_effect = {
        add_estate_loyalty_modifier = {
            estate = estate_cossacks
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}
