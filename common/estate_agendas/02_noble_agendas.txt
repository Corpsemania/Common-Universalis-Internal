#Get allies
estate_nobles_get_allies = {
	can_select = {
		NOT = { num_of_allies = 3 }
		NOT = { num_of_subjects = 3 }
		is_at_war = no
		num_of_free_diplomatic_relations = 1
		NOT = { has_country_modifier = foreign_contacts }
	}
	selection_weight = {
		factor = 1
		modifier = {
			factor = 2
			NOT = { num_of_allies = 1 }
			NOT = { num_of_subjects = 1 }
		}
	}
	pre_effect = {
		set_variable = {
			which = estate_nobles_get_allies_variable
			value = 0
		}
		if = {
			limit = {
				OR = {
					NOT = { num_of_allies = 1 }
					AND = {
						num_of_subjects = 2
						NOT = { num_of_allies = 2 }
					}
					AND = {
						NOT = { num_of_allies = 2 }
						NOT = { num_of_free_diplomatic_relations = 2 }
					}
				}
			}
			set_variable = {
				which = estate_nobles_get_allies_variable
				value = 1
			}
		}
		else = {
			set_variable = {
				which = estate_nobles_get_allies_variable
				value = 2
			}
		}
	}
	fail_if = {
		is_free_or_tributary_trigger = no
	}
	task_requirements = {
		if = {
			limit = {
				check_variable = {
					which = estate_nobles_get_allies_variable
					value = 2
				}
			}
			num_of_allies = 3
		}
		else = {
			num_of_allies = 2
		}
	}
	task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = 10
		}
		add_country_modifier = {
			name = foreign_contacts
			duration = 3650
		}
	}
    failing_effect = {
        add_estate_loyalty_modifier = {
            estate = estate_nobles
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}

		prestige = -30
		NOT = { prestige = 20 }
	}
	selection_weight = {
		factor = 2.5
		modifier = {
			factor = 2
			NOT = {
				prestige = 0
			}
		}
	}
	pre_effect = {
		set_variable = {
			which = estate_nobles_improve_prestige_var
			value = 0
		}
		if = {
			limit = {
				NOT = {
					prestige = -10
				}
			}
			set_variable = {
				which = estate_nobles_improve_prestige_var
				value = 1
			}
		}

			limit = {
				NOT = {
					prestige = 15
				}
			}
			set_variable = {
				which = estate_nobles_improve_prestige_var
				value = 2
			}
		}
		else = {
			set_variable = {
				which = estate_nobles_improve_prestige_var
				value = 3
			}
		}
	}
	task_requirements = {
		if = {
			limit = {
				check_variable = {
					which = estate_nobles_improve_prestige_var
					value = 3
				}
			}
			prestige = 50
		}
		else_if = {
			limit = {
				check_variable = {
					which = estate_nobles_improve_prestige_var
					value = 2
				}
			}
			prestige = 30
		}
		else = {
			prestige = 10
		}
	}
	task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_nobles

		}
		add_adm_power = 50
	}
    failing_effect = {
