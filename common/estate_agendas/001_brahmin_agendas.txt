#Convert a province
estate_brahmins_convert_province_x = {
	can_select = {

		num_of_missionaries = 1
		any_owned_province = {
			is_state_core = root
			NOT = { religion = root }
			NOT = { has_province_modifier = religious_zeal_at_conv }
			NOT = { has_province_modifier = expelled_minority }
			NOT = { has_province_modifier = religious_migration }
			NOT = { religion = secondary }
			has_missionary = no
			missionary_strength = 0.01
		}
	}
	selection_weight = {
		factor = 1
		modifier = {
			factor = 1.5
			NOT = { religious_unity = 0.85 }
		}
		modifier = {
			factor = 1.5
			NOT = { religious_unity = 0.75 }
		}
		modifier = {
			factor = 1.5
			NOT = { religious_unity = 0.65 }
		}
		modifier = {
			factor = 1.5
			NOT = { religious_unity = 0.5 }
		}
		modifier = {
			factor = 1.5
			NOT = { religious_unity = 0.4 }
		}
		modifier = {
			factor = 0.25
			religion = confucianism
		}
		modifier = {
			factor = 0.35
			has_idea_group = humanist_ideas
		}
		modifier = {
			factor = 2
			has_idea_group = religious_ideas
		}
		modifier = {
			factor = 0.75
			NOT = { num_of_missionaries = 2 }
		}
		modifier = {
			factor = 1.25
			is_defender_of_faith = yes
		}
		modifier = {
			factor = 0.35
			has_country_modifier = recent_converted_province_mission_timer
		}
	}
	pre_effect = {
		random_owned_province = {
			limit = {
				is_state_core = root
				NOT = { religion = root }
				NOT = { has_province_modifier = religious_zeal_at_conv }
				NOT = { has_province_modifier = expelled_minority }
				NOT = { has_province_modifier = religious_migration }
				NOT = { religion = secondary }
				has_missionary = no
			}
			save_event_target_as = agenda_province
		}
		random_owned_province = {
			limit = {
				is_state_core = root
				NOT = { religion = root }
				NOT = { has_province_modifier = religious_zeal_at_conv }
				NOT = { has_province_modifier = expelled_minority }
				NOT = { has_province_modifier = religious_migration }
				NOT = { religion = secondary }
				has_missionary = no
				development = 10
			}
			save_event_target_as = agenda_province
		}
	}
	provinces_to_highlight = {
		province_id = event_target:agenda_province
	}
	fail_if = {
		event_target:agenda_province = {
			NOT = { owned_by = root }
		}
	}
	immediate_effect = {
		event_target:agenda_province = {
			add_province_modifier = {
				name = agenda_conversion_zeal
				duration = 7300
			}
		}
	}
	on_invalid = {
		event_target:agenda_province = {
			if = {
				limit = { has_province_modifier = agenda_conversion_zeal }
				remove_province_modifier = agenda_conversion_zeal
			}
		}
	}
	task_requirements = {
		event_target:agenda_province = {
			religion = root
		}
	}
    task_completed_effect = {
		hidden_effect = {
			event_target:agenda_province = {
				if = {
					limit = { has_province_modifier = agenda_conversion_zeal }
					remove_province_modifier = agenda_conversion_zeal
				}
			}
			add_country_modifier = {
				name = recent_converted_province_mission_timer
				duration = 3650
				hidden = yes
			}
		}
		generate_scaled_advisor_of_type_and_religion_effect = {
			type = inquisitor
			religion = hinduism
			discount = yes
		}
		add_estate_loyalty = {
			estate = estate_brahmins
			loyalty = 10
		}
    }
    failing_effect = {
		event_target:agenda_province = {
			if = {
				limit = { has_province_modifier = agenda_conversion_zeal }
				remove_province_modifier = agenda_conversion_zeal
			}
		}
        add_estate_loyalty_modifier = {
            estate = estate_brahmins
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}

#Get allies
estate_brahmins_get_allies = {
	can_select = {
		religion = hinduism
		NOT = { num_of_allies = 3 }
		NOT = { num_of_subjects = 3 }
		is_at_war = no
		num_of_free_diplomatic_relations = 1
		NOT = { has_country_modifier = foreign_contacts }
	}
	selection_weight = {
		factor = 1
		modifier = {
			factor = 2
			NOT = { num_of_allies = 1 }
			NOT = { num_of_subjects = 1 }
		}
	}
	pre_effect = {
		set_variable = {
			which = estate_brahmins_get_allies_variable
			value = 0
		}
		if = {
			limit = {
				OR = {
					NOT = { num_of_allies = 1 }
					AND = {
						num_of_subjects = 2
						NOT = { num_of_allies = 2 }
					}
					AND = {
						NOT = { num_of_allies = 2 }
						NOT = { num_of_free_diplomatic_relations = 2 }
					}
				}
			}
			set_variable = {
				which = estate_brahmins_get_allies_variable
				value = 1
			}
		}
		else = {
			set_variable = {
				which = estate_brahmins_get_allies_variable
				value = 2
			}
		}
	}
	fail_if = {
		is_free_or_tributary_trigger = no
	}
	task_requirements = {
		if = {
			limit = {
				check_variable = {
					which = estate_brahmins_get_allies_variable
					value = 2
				}
			}
			num_of_allies = 3
		}
		else = {
			num_of_allies = 2
		}
	}
	task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_brahmins
			loyalty = 10
		}
		add_country_modifier = {
			name = foreign_contacts
			duration = 3650
		}
	}
    failing_effect = {
        add_estate_loyalty_modifier = {
            estate = estate_brahmins
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}
