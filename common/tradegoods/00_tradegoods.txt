# ### GOODS TYPES ### 
#
#	> Deposit 
#		"Extracted with mines or quaries, their size connected to the production level"
#		* +50% local_production_efficiency <OR> -20% local_development_cost (gold)
#
#	> Commodity 
#		"Produced mainly for trade, stimulating local commerce"
#		* -10% production_efficiency
#		* +10% local_tax_modifier
#
#	> Food
#		"Goods mostly used to feed the population"
#		* -5% local_development_cost
#		* +10% local_manpower_modifier (plants) <OR> +20% local_sailors_modifier (aquatic) <OR> +20% local_friendly_movement_speed (livestock)
#		* +10 local_colonial_growth
#
#	> Forestry
#		"Poor economies relying on forests"
#		* -10% local_tax_modifier
#		* -5% local_development_cost
#		* -10% local_build_cost
#		* -0.05 monthly_devastation
#
#	> Limited
#		"Goods native to the province, gathered or hunted without direct means to increase the production"
#		[furs (2), amber (3), ivory (3), gems (4), jade (4), pearls (4)]
#		* +X flat trade_value
#		* -90% trade_goods_size_modifier
#		* +10% local_tax_modifier
#
#	> Urban
#		"Represents concentrated urban production relying on infrastructure"
#		? global modifiers per province
#		* +10% local_tax_modifier
#		* +10% province_trade_power_modifier
#		* +50% local_production_efficiency
#		
# ### URBAN GOODS ###
#
#	Required rural good, geographic area or technology.
#	Rural goods must be produced by province itself or one of its neighbors.
#
#	Beer		[grain]
#	Tools		[iron]
#	Bronze		[copper, Bronze Cannons (16th century event)
#	Leather		[livestock]
#	Naval 		[coast, lumber]
#	Furniture	[lumber, tools]
#	Jewelry		[gems, amber, jade]
#	Coins		[gold, silver]
#	Rum			[sugar, New World]	
#	Chinaware	[Far East]
#	Silk		[Silk Road]
#	Weapons		[tools, Arquebus (Military Tech 6), innovativeness]
#	Paper		[secondary good - paper <OR> cloth]
#	Glass		[secondary good - glass <OR> Art Power]
#	Cloth		[default]
#
