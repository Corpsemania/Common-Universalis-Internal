			limit = {
				OR = {
					AND = {
						has_country_flag = kbo_gazargamu
						NOT = { has_country_modifier = kbo_religious_orthodoxy }
						NOT = { has_country_modifier = kbo_death_to_kingmakers }
					}

#Foundation of Gazargamu
country_event = {
	id = flavor_kbo.2
	title = flavor_kbo.2.t
	desc = flavor_kbo.2.d
	picture = CITY_VIEW_eventPicture

	fire_only_once = yes

	trigger = {
		tag = KBO
		has_country_flag = kbo_two_branches
		NOT = { is_year = 1550 }
		owns = 1157
	}
	
	mean_time_to_happen = {
		months = 318
		modifier = {
			factor = 0.8
			stability = 2
		}
		modifier = {
			factor = 0.8
			stability = 3
		}
		modifier = {
			factor = 0.8
			years_of_income = 0.4
		}
	}
	
	option = {
		name = flavor_kbo.2.a	#Build Gazargamu
		ai_chance = { factor = 1 }
		if = {
			limit = {
				NOT = { capital = 1157 }
			}
			set_capital = 1157
			hidden_effect = { cu_calc_country_absolute_dist = yes }
		}
		add_years_of_income = -0.3
		add_adm_power = -25
		set_country_flag = kbo_gazargamu
		1157 = {
			add_province_modifier = {
			name = kbo_gazargamu
			duration = 7300
			}
		}
	}
	option = {
		name = flavor_kbo.2.b	#Our current capital is fine.
		ai_chance = { factor = 0 }
		cu_change_stability_point_p50 = yes
	}
}

		tag = KBO

		government = monarchy
	}
	
	mean_time_to_happen = {
		months = 318
		modifier = {
			factor = 0.8
			stability = 2
		}
		modifier = {
			factor = 0.8
			stability = 3
		}
		modifier = {
			factor = 0.6
			religious_unity = 1.0
		}
	}
	
	option = {
		name = flavor_kbo.3.a	#No more syncretism!

		cu_change_stability_point_n75 = yes

			name = kbo_religious_orthodoxy
			duration = -1
		}
	}
	
	option = {
		name = flavor_kbo.3.b	#Let us not go overboard with this.
		ai_chance = { factor = 0 }
		cu_change_stability_point_p50 = yes
	}
}
