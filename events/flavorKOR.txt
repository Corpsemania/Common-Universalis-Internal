# Hyo-jong's northern expedition
country_event = {
	id = flavor_kor.1
	title = "flavor_kor.EVTNAME1"
	desc = "flavor_kor.EVTDESC1"
	picture = MILITARY_CAMP_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = KOR		
		is_year = 1600
		NOT = { is_year = 1700 }
		any_owned_province = {
			any_neighbor_province = {
				primary_culture = manchu
				primary_culture = manchu_han
			}
		}
	}
	
	mean_time_to_happen = {
		months = 200
	}
	
	option = {
		name = "flavor_kor.EVTOPTA1"		
		if = {
			limit = {
				NOT = {
					any_owned_province = {
						any_neighbor_province = {
							owner = {
								OR = {
									primary_culture = manchu
									primary_culture = manchu_han
								}
							}
							NOT = { is_claim = root }
						}
					}
				}
			}
			add_country_modifier = {
				name = kor_manchurian_expedition
				duration = 3650
			}
		}
		else = {
			every_neighbor_country = {
				limit = {
					OR = {
						primary_culture = manchu
						primary_culture = manchu_han
					}
				}
				every_owned_province = {
					limit = {
						NOT = { is_claim = root }
						any_neighbor_province = {
							owned_by = root
						}
					}
					add_claim = ROOT
				}
			}
		}
	}
}

#Im Kkeokjeong's Rebellion
country_event = {
	id = flavor_kor.9
	title = flavor_kor.9.t
	desc = flavor_kor.9.d
	picture = STREET_SPEECH_eventPicture

	fire_only_once = yes
	mean_time_to_happen = { days = 7300 }
	
	trigger = {
		tag = KOR
		is_year = 1550
		has_country_flag = kor_peasant_struggles
		NOT = { has_country_flag = kor_peasant_struggles_resolved }
	}

	immediate = {
		hidden_effect = {
			if = {
				limit = {
					any_owned_province = {
						OR = {
							province_id = 4231
							province_id = 733
						}
						is_capital = no
					}
				}
				random_owned_province = {
					limit = {
						OR = {
							province_id = 4231
							province_id = 733
						}
						is_capital = no
					}
					save_event_target_as = ImKkeokjeong_Revolt_Province
				}
			}
			else_if = {
				limit = {
					any_owned_province = {
						culture = korean
						NOT = { province_id = 2741 }
						is_capital = no
					}
				}
				random_owned_province = {
					limit = {
						culture = korean
						NOT = { province_id = 2741 }
						is_capital = no
					}
					save_event_target_as = ImKkeokjeong_Revolt_Province
				}
			}
			else_if = {
				limit = {
					any_owned_province = {
						culture = korean
					}
				}
				random_owned_province = {
					limit = {
						culture = korean
					}
					save_event_target_as = ImKkeokjeong_Revolt_Province
				}
			}
			else = {
				random_owned_province = {
					save_event_target_as = ImKkeokjeong_Revolt_Province
				}
			}
		}
	}

	option = {
		name = flavor_kor.9.opt
		remove_country_modifier = kor_peasant_struggles

		event_target:ImKkeokjeong_Revolt_Province = {
			spawn_rebels = {
				size = 2
				type = leadered_peasant_rebels
				leader = "Im Kkeokjeong"
				win = yes
			}
		}
	}
}
