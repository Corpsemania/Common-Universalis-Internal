		owner = { has_country_flag = art_transfer_ing }
		has_province_flag = art_transfer_dest

		}

			owner = {

					limit = { has_province_flag = art_transfer_source }

						limit = { has_province_flag = art_transfer_1 } 
						change_variable = { which = cu_art_piece1 value = -1 }

					}
					else = {
						change_variable = { which = cu_art_piece0 value = -1 } 
						clr_province_flag = art_transfer_0
					}
					clr_province_flag = art_transfer_source
				}
			}
		}
	}
	  
	option = { 
		name = "urban.71.a"
		owner = {
			clr_country_flag = art_transfer_ing
		}
		if = {
			limit = { has_province_flag = art_transfer_2 }

				change_variable = { which = cu_art_slot value = -1 }

				change_variable = { which = cu_art_slot value = -1 }
				change_variable = { which = cu_art_piece value = 1 }
				change_variable = { which = cu_art_piece1 value = 1 }
			}
			custom_tooltip = urban.71.at1
		}
		else = {
			hidden_effect = {
				change_variable = { which = cu_art_slot value = -1 }
				change_variable = { which = cu_art_piece value = 1 }
				change_variable = { which = cu_art_piece0 value = 1 }
			}
			custom_tooltip = urban.71.at0
		}
		hidden_effect = {
		owner = { cu_display_cultural_power = yes }
		}
	}
