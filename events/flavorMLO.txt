# The Ambrosian Republic
country_event = {
	id = flavor_mlo.1
	title = flavor_mlo.1.t
	desc = flavor_mlo.1.d
	picture = DEBATE_REPUBLICAN_eventPicture
	
	fire_only_once = yes
	is_triggered_only = yes

	trigger = {
		tag = MLO
		government = monarchy
		NOT = { has_reform =  celestial_empire }
		OR = {
			has_regency = yes
			NOT = { legitimacy = 75 }
		}
		is_lesser_in_union = no
		NOT = {
			is_year = 1550
		}
	}
	
	option = {
		name = flavor_mlo.1.a
		ai_chance = { factor = 90 }
		add_government_reform = becomes_republic_reform		
		add_republican_tradition = 5
		every_neighbor_country  = {
			limit = {
				is_subject = no
				government = monarchy
				religion = catholic
			}
			add_opinion = {
				who = ROOT
				modifier = mlo_against_ambrosian
			}
			if = {
				limit = {
					government = monarchy
					NOT = { has_reform =  elective_monarchy }
				}
				
				add_casus_belli = {	
					target = MLO
					type = cb_change_government
					months = 120
				}		
			}
		}
		hidden_effect = {
			if = {
				limit = {
					has_regency = yes
				}
				country_event = { id = 700 }
			}
		}
	}
	option = {
		name = flavor_mlo.1.b
		cu_change_stability_point_n150 = yes
		add_legitimacy = 25
		add_reform_desire = 0.01

	}
}

# Niccolo Macchiavelli
