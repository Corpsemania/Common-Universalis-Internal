#Tensions between Nomads and Sedentary Population
country_event = {
	id = tribal_estate_events.1
	title = tribal_estate_events.1.t
	desc = tribal_estate_events.1.d
	picture = NOMAD_ESTATE_eventPicture
	
	trigger = {
		has_dlc = "The Cossacks"
		has_estate = estate_nomadic_tribes
		num_of_cities = 5
		any_owned_province = {
			has_estate = estate_nomadic_tribes
			can_spawn_rebels = nationalist_rebels
		}
	}

	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					has_estate = estate_nomadic_tribes
					can_spawn_rebels = nationalist_rebels
				}
				area = {
					limit = {
						can_spawn_rebels = nationalist_rebels
						owned_by = ROOT
					}
					set_province_flag = conflict_regarding_tax_collection
				}
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = tribal_estate_events.1.a #	Reprimand the Chiefs.
		ai_chance = {
			factor = 75
			modifier = {
				factor = 0.5
				NOT = {
					estate_loyalty = {
						estate = estate_nomadic_tribes
						loyalty = 40
					}
				}
			}
			modifier = {
				factor = 1.5
				estate_influence = {
					estate = estate_nomadic_tribes
					influence = 65
				}
			}
		}
		add_estate_loyalty = {
			estate = estate_nomadic_tribes
			loyalty = -15
		}
		add_estate_influence_modifier = {
			estate = estate_nomadic_tribes
			desc = EST_VAL_TRIBES_REPRIMANDED
			influence = -10
			duration = 5475
		}
	}
	option = {
		name = tribal_estate_events.1.b #	Let it Slide
		ai_chance = {
			factor = 25
			modifier = {
				factor = 1.5
				NOT = {
					estate_loyalty = {
						estate = estate_nomadic_tribes
						loyalty = 40
					}
				}
			}
			modifier = {
				factor = 0.5
				estate_influence = {
					estate = estate_nomadic_tribes
					influence = 65
				}
			}
		}
		random_owned_province = {
			limit = {
				has_province_flag = conflict_regarding_tax_collection
			}
			spawn_rebels = {
				size = 1
				type = nationalist_rebels
			}
			clr_province_flag = conflict_regarding_tax_collection
		}
		random_owned_province = {
			limit = {
				has_province_flag = conflict_regarding_tax_collection
				owner = { num_of_cities = 10 }
			}
			spawn_rebels = {
				size = 1
				type = nationalist_rebels
			}
			clr_province_flag = conflict_regarding_tax_collection
		}
		random_owned_province = {
			limit = {
				has_province_flag = conflict_regarding_tax_collection
				owner = { num_of_cities = 15 }
			}
			spawn_rebels = {
				size = 1
				type = nationalist_rebels
			}
			clr_province_flag = conflict_regarding_tax_collection
		}
		every_owned_province = {
			clr_province_flag = conflict_regarding_tax_collection
		}
		add_estate_loyalty = {
			estate = estate_nomadic_tribes
			loyalty = 15
		}
		add_estate_influence_modifier = {
			estate = estate_nomadic_tribes
			desc = EST_VAL_TRIBES_ALLOWED_FREE_REIGN
			influence = 10
			duration = 5475
		}
	}
}
