# Local Governor enacts trade trade laws
country_event = {
	id = china_events.1
	title = "china_events.1.t"
	desc = "china_events.1.d"
	picture = ADVISOR_eventPicture
	
	trigger = {
		any_owned_province = {
			is_capital = no
			cu_not_terr_trigger = yes
			NOT = { has_province_modifier = chi_local_trade_regulations }
			NOT = { has_province_modifier = chi_local_trade_down_turn }
		}
		OR = {
			faction_in_power = enuchs
			AND = {
				has_dlc = "Mandate of Heaven"
				is_emperor_of_china = yes
			}
		}
	}
	
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			random_owned_province = { #Atleast here
				limit = {
					is_capital = no
					cu_not_terr_trigger = yes
					NOT = { has_province_modifier = chi_local_trade_regulations }
					NOT = { has_province_modifier = chi_local_trade_down_turn }
				}
				save_event_target_as = trade_problems_province
			}
			random_owned_province = { #But rather here
				limit = {
					is_capital = no
					has_building = marketplace
					NOT = { has_province_modifier = chi_local_trade_regulations }
					NOT = { has_province_modifier = chi_local_trade_down_turn }
				}
				save_event_target_as = trade_problems_province
			}
			random_owned_province = { #Here would be even better
				limit = {
					is_capital = no
					has_port = yes
					NOT = { has_province_modifier = chi_local_trade_regulations }
					NOT = { has_province_modifier = chi_local_trade_down_turn }
				}
				save_event_target_as = trade_problems_province
			}
			random_owned_province = { #Warmer
				limit = {
					is_capital = no
					has_port = yes
					has_building = marketplace
					NOT = { has_province_modifier = chi_local_trade_regulations }
					NOT = { has_province_modifier = chi_local_trade_down_turn }
				}
				save_event_target_as = trade_problems_province
			}
			random_owned_province = { #Best
				limit = {
					is_capital = no
					has_province_modifier = center_of_trade_modifier
					NOT = { has_province_modifier = chi_local_trade_regulations }
					NOT = { has_province_modifier = chi_local_trade_down_turn }
				}
				save_event_target_as = trade_problems_province
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "china_events.1.a"
		boost_eunuchs_effect = yes
		add_meritocracy = -5
		if = {
			limit = {
				NOT = { has_dlc = "Mandate of Heaven" }
			}
			add_legitimacy = -5
		}
		event_target:trade_problems_province = {
			add_province_modifier = {
				name = "chi_local_trade_regulations"
				duration = 3650
			}
		}
	}
	
	option = {
		name = "china_events.1.b"
		reduce_eunuchs_effect = yes
		add_meritocracy = 5
		event_target:trade_problems_province = {
			add_province_modifier = {
				name = "chi_local_trade_down_turn"
				duration = 3650
			}
		}
	}
}

# 'When the people have enough, how can the ruler alone have too little?'
# Based on Wang Ao's 8-legged essay.
country_event = {
	id = china_events.16
	title = "china_events.16.t"
	desc = "china_events.16.d"
	picture = UNIVERSITY_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		NOT = { has_country_modifier = chi_efficient_taxation }
		OR = {
			AND = {
				faction_in_power = bureaucrats
			}
			AND = {
				has_dlc = "Mandate of Heaven"
				is_emperor_of_china = yes
				meritocracy = 70
			}
		}
		stability = 1
		any_owned_province = {
			cu_not_terr_trigger = yes
			NOT = { devastation = 1 }
			NOT = { has_province_modifier = chi_lowered_taxes_in_state }
		}
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					cu_not_terr_trigger = yes
					NOT = { devastation = 1 }
					is_prosperous = no
					NOT = { has_province_modifier = chi_lowered_taxes_in_state }
				}
				save_event_target_as = prosperity_boost_here
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "china_events.16.a"
		boost_bureaucrats_effect = yes
		if = {
			limit = { has_dlc = "Mandate of Heaven" }
			event_target:prosperity_boost_here = {
				if = {
					limit = { NOT = { prosperity = 100 } }
					add_prosperity = 25
				}
				area = {
					add_province_modifier = {
						name = "chi_lowered_taxes_in_state"
						duration = 1825
					}
				}
			}
		}
		if = {
			limit = { NOT = { has_dlc = "Mandate of Heaven" } }
			cu_change_stability_point_p50 = yes
		}
	}
	
	option = {
		name = "china_events.16.b"
		reduce_bureaucrats_effect = yes
		reduce_meritocracy_effect = yes
		add_country_modifier = {
			name = "chi_efficient_taxation"
			duration = 3650
