#Cabinet Main Panel
country_event = {
	id = cabinet.0
	title = cabinet.0.t
	desc = cabinet.0.d
	picture = ADVISOR_eventPicture
    
    is_triggered_only = yes
	
	immediate = {
	    hidden_effect = {
			if = {
				limit = { NOT = { has_country_flag = cabinet_inialized } }
				#Initial Setup
				set_country_flag = cabinet_ep_state
				set_country_flag = cabinet_ep_primary
				set_country_flag = cabinet_inialized
			}
		}
	}
    
	option = {
	    name = "cabinet.0.census" #Country Census
		hidden_effect = {
		    cu_country_census = yes
			cu_calc_total_art_solts = yes				
		    country_event = { id = cabinet.99 }
			set_country_flag = cu_stability_fixed
			cu_calc_total_stability_growth = yes
			clr_country_flag = cu_stability_fixed
		}
	}
	
	option = {
	    name = "cabinet.0.culture" #Cultural Edicts
		hidden_effect = {
		    country_event = { id = cabinet.3 }
		}
	}
	
	option = {
		trigger = {
			NOT = { 
				OR = {
					has_reform = steppe_horde
					has_reform = shogunate
					has_reform = daimyo
					has_reform = indep_daimyo
				}
			} 
		}	
	    name = "cabinet.0.economy" #Economic Edicts
		hidden_effect = {
		    country_event = { id = cabinet.4 }
		}
	}
	
	option = {
	    name = "cabinet.0.PAC" #PAC settings
		hidden_effect = {
		    country_event = { id = cabinet.5 }
		}
	}
	
	option = { 
		name = "cabinet.0.expend" #Exert Now
		custom_tooltip = cabinet.expend.t
		cu_do_cabinet_country = yes
		clr_country_flag = cabinet_panel_open
	}
	option = { 
		name = "cabinet.0.core" #Revise integration policy
		custom_tooltip = cabinet.core.t
		hidden_effect = {
		     every_owned_province = {
			      limit = {
				       has_province_modifier = cu_local_autonomy_administration
				  }
				  remove_province_modifier = cu_local_autonomy_administration
				  set_province_flag = cu_newly_conquered_province
			 }
		}
	}

    option = {
	    name = "cu.exit"
		clr_country_flag = cabinet_panel_open
	}
}
