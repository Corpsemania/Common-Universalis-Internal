# Rajputs
country_event = {
	id = india_mission_events.1
	title = "india_mission_events.1.t"
	desc = "india_mission_events.1.d"
	picture = 4_Rajput_Estate_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		has_dlc = "Dharma"
	}
	mean_time_to_happen = {
		days = 1
	}
	option = { # favour rajput nobility
		name = "india_mission_events.1.a"
		trigger = {
			has_estate = estate_rajput
		}
		add_estate_loyalty_modifier = {
			estate = estate_rajput
			desc = EST_VAL_RAJPUT_DOMINANCE
			loyalty = 10
			duration = -1
		}
	}
	option = { # favour local nobles
		trigger = {
			NOT = {
				has_estate = estate_rajput
			}
		}
		name = "india_mission_events.1.c"
		add_stability_or_adm_power = yes
		west_india_region = {
			limit = {
				owned_by = ROOT
			}
			add_province_modifier = {
				name = "hin_rajput_integration"
				duration = 7300
			}
		}
	}
}

	}

			add_base_manpower = 1
			add_province_modifier = {
				name = "peacock_throne"
				duration = -1
			}
		}
	}
}

# Generic Indian mission events

country_event = {
	id = india_mission_events.5
	title = "india_mission_events.5.t"
	desc = {
		trigger = {
			NOT = {
				primary_culture = marathi
			}
		}
		desc = "india_mission_events.5.d"
	}
	desc = {
		trigger = {
			primary_culture = marathi
		}
		desc = "india_mission_events.5.d2"
	}
	picture = 13_Maratha_Estate_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		has_dlc = "Dharma"
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = { # For Marathas
		name = EXCELLENT
		trigger = {
			primary_culture = marathi
		}
		add_army_tradition = 20
	}
	option = { # Adopt Marathi culture
		name = "india_mission_events.5.b"
		trigger = {
			NOT = {
				primary_culture = marathi
			}
		}
		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				OR = {
					tag = VIJ
					tag = BAH
				}
			}
		}
		change_primary_culture = marathi
		hidden_effect = {
			kill_heir = yes
		}
		change_variable = { which = cu_var1 value = -100 }
		cu_change_stability_point = yes
		define_ruler = {
			culture = marathi
			MIL = 4
		}
		if = {
			limit = {
				has_states_general_mechanic = yes
			}
			change_statists_vs_orangists = 1
		}
	}
	option = { # Keep current culture
		name = "india_mission_events.5.c"
		trigger = {
			NOT = {
				primary_culture = marathi
			}
		}
		ai_chance = {
			factor = 100
			modifier = {
				factor = 0
				NOT = { tag = VIJ }
				NOT = { tag = BAH }
			}
		}
		if = {
			limit = {
				has_estate = estate_maratha
			}
			add_estate_loyalty = {
				estate = estate_maratha
				loyalty = 20
			}
		}
	}
}
