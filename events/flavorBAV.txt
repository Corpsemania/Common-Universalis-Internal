# Opposition against Austria
country_event = {
	id = flavor_bav.3
	title = flavor_bav.3.t
	desc = flavor_bav.3.d
	picture = COURT_eventPicture
	
	fire_only_once = yes

	trigger = {
		tag = BAV
		
		is_year = 1450
		NOT = { is_year = 1500 }
		is_lesser_in_union = no
		dynasty = "von Wittelsbach"
		
		HAB = {
			exists = yes
			is_emperor = yes
			dynasty =  "von Habsburg"
		}
	}
	
	mean_time_to_happen = {
		months = 120
	}	
	
	
	option = {	
		name = flavor_bav.3.a	
		HAB = {
			add_opinion = {
				who = ROOT
				modifier = bav_against_habsburg
			}
		}
		add_legitimacy = 5
		add_prestige = 5
	}

	option = {	
		name = flavor_bav.3.b	
		HAB = {
			add_opinion = {
				who = ROOT
				modifier = bav_reconcile_habsburg
			}
		}
		add_prestige = -25
	}		
}

#Wilhelm IV invites the Jesuits into Bavaria
country_event = {
	id = flavor_bav.5
	title = flavor_bav.5.t
	desc = flavor_bav.5.d
	picture = RELIGION_eventPicture
	
	fire_only_once = yes

	trigger = {
		tag = BAV

		is_year = 1520
		NOT = { is_year = 1560 }
		is_religion_enabled = protestant
		religion = catholic
		dominant_religion = catholic
		has_country_modifier = counter_reformation
	}
	
	mean_time_to_happen = {
		months = 120
	}	
	
	
	option = {	
		name = flavor_bav.5.a
		add_papal_influence = 10
		add_country_modifier = {
			name = "bavarian_jesuits"
			duration = -1
		}
	}	
	option = {	
		name = flavor_bav.5.b
		trigger = {
			exists = PAP
		}
		PAP = {
			add_opinion = {
				who = ROOT
				modifier = bav_jesuits_rejected
			}
		}
		add_papal_influence = -10
	}		
}
