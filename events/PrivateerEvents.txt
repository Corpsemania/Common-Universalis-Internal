# Local Traders Demand Aid - privateered province demands funds for protection
province_event = {
	id = privateers.1
	title = "privateers.1.t"
	desc = "privateers.1.d"
	picture = HARBOUR_eventPicture

	trigger = {
                has_port = yes
		privateer_power = {
			share = 20
		}
	}

	mean_time_to_happen = {
		months = 5000
		modifier = {
			factor = 0.8
			owner = { has_reform = merchants_reform }
		}
		modifier = {
			factor = 0.8
			privateer_power = { share = 40 }
		}
		modifier = {
			factor = 1.5
			owner = {
				num_of_cities = 10
			}
		}
	}

	# Give them money to build arm their ships for defense
	option = {
		name = "privateers.1.a"
		owner = { add_treasury = -100 }
	}

	# Refuse to help, reducing local trade power
	option = {
		name = "privateers.1.b"
		add_province_modifier = {
			name = "refused_aid_traders"
			duration = 1825
		}
	}
}

## Invalid Letter of Marque? - colonial governor has issued letter for rivateer
# Version of the event that hits another nation, not the overlord
country_event = {
	id = privateers.2
	title = "privateers.2.t"
	desc = "privateers.2.d"
	picture = DEBATE_REPUBLICAN_eventPicture

	trigger = {
		NOT = { has_reform = colonial_government }
		any_active_trade_node = {
			any_privateering_country = {
				has_reform = colonial_government
				NOT = {
					colonial_parent = { tag = ROOT }
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 480
	}

	immediate = {
		random_active_trade_node = {
			limit = {
				privateer_power = { share = 1 }
			}
			random_privateering_country = {
				limit = {
					has_reform = colonial_government
				}
				set_country_flag = invalid_letter_marque
			}
		}
	}

	option = {
		name = "privateers.2.a"
		random_country = {
			limit = {
				has_country_flag = invalid_letter_marque
			}
			add_opinion = {
				who = ROOT
				modifier = opinion_insulted
			}
			clr_country_flag = invalid_letter_marque
		}
	}

	option = {
		name = "privateers.2.b"
		random_country = {
			limit = {
				has_country_flag = invalid_letter_marque
			}
			add_opinion = {
				who = ROOT
				modifier = opinion_pleased
			}
			clr_country_flag = invalid_letter_marque
		}
		add_prestige = -10
	}
}

# Oceans Beyond Privateering - privateer wishes to become explorer
country_event = {
	id = privateers.6
	title = "privateers.6.t"
	desc = "privateers.6.d"
	picture = SHIP_SAILING_eventPicture

	trigger = {
		has_privateers = yes
		NOT = { has_country_flag = funded_explorer_privateer }
		any_trade_node = {
			privateer_power = {
				country = ROOT
				share = 10
			}
			is_overseas = yes
		}
	}

	mean_time_to_happen = {
		months = 720
		modifier = {
			factor = 0.7
			has_idea_group = exploration_ideas
		}
	}

	# Fund expedition
	option = {
		name = "privateers.6.a"
		add_treasury = -100
		set_country_flag = funded_explorer_privateer
		if = {
			limit = {
				is_year = 1570
				NOT = { is_year = 1600 }
				OR = {
					tag = ENG
					tag = GBR
				}
			}
			define_explorer = {
				name = "Francis Drake"
				fire = 3
				shock = 3
				manuever = 5
				siege = 0
			}
		}
		if = {
			limit = {
				NOT = {
					OR = {
						tag = ENG
						tag = GBR
					}
				}
			}
			create_explorer = { tradition = 5 }
		}
	}

	# Tell the privateer to stay where he is
	option = {
		name = "privateers.6.b"
		add_prestige = -5
	}
}

# Captured Slavers - privateers capture a ship transporting slaves
country_event = {
	id = privateers.9
	title = "privateers.9.t"
	desc = "privateers.9.d"
	picture = MERCHANTS_TALKING_eventPicture

	trigger = {
		has_privateers = yes
		has_country_modifier = the_abolish_slavery_act
		any_trade_node = {
			privateer_power = {
				country = ROOT
				share = 1
			}
			any_trade_node_member_province = {
				trade_goods = slaves
			}
		}
	}

	mean_time_to_happen = {
		months = 600
	}

	# Treat the slavers as pirates
	option = {
		name = "privateers.9.a"
		add_dip_power = 20
	}

	# Treat the slavers as foreign traders
	option = {
		name = "privateers.9.b"
		random_trade_node = {
			limit = {
				privateer_power = {
					country = ROOT
					share = 1
				}
				any_trade_node_member_province = {
					trade_goods = slaves
				}
			}
			every_trade_node_member_country = {
				limit = {
					NOT = { has_country_modifier = the_abolish_slavery_act }
				}
				add_opinion = {
					who = ROOT
					modifier = opinion_pleased
				}
			}
		}
	}
}

	id = privateers.13

	desc = "privateers.13.d"
	picture = ACCUSATION_eventPicture

	is_triggered_only = yes

	trigger = {
		has_country_flag = promised_merchants_war
	}

	option = {
		name = "privateers.13.a"
		add_prestige = 5
		hidden_effect = {
			country_event = {
				id = privateers.14
				days = 730
			}
		}
	}

	option = {
		name = "privateers.13.b"
		cu_change_stability_point_n75 = yes
		remove_country_modifier = merchants_prepare_war
		clr_country_flag = promised_merchants_war
	}

	option = {
		trigger = {
			NOT = {
				any_active_trade_node = {
					trade_share = {
						country = ROOT
						share = 10
					}
					privateer_power = {
						share = 1
					}
				}
			}
		}
		name = "privateers.13.c"
		remove_country_modifier = merchants_prepare_war
		clr_country_flag = promised_merchants_war
	}
}
