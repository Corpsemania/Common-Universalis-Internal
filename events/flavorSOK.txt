namespace = flavor_sok
normal_or_historical_nations = yes

#Usman dan Fodio
country_event = {
	id = flavor_sok.1
	title = flavor_sok.1.t
	desc = flavor_sok.1.d
	picture = RELIGIOUS_TURMOIL_eventPicture

	fire_only_once = yes

	trigger = {
		owns = 1148 #Gobir
		is_year = 1770
		OR = {
			NOT = { religion = sunni }
			NOT = { piety = -0.3 }
		}
		any_known_country = {
			has_country_flag = fulani_jihad_state
		}
	}
	
	mean_time_to_happen = {
		months = 150
	}
	
	option = {
		name = flavor_sok.1.a	#Accept the invitation of dan Fodio
		ai_chance = { factor = 1 }
		trigger = {
			capital_scope = {
				OR = {
					area = hausa_area
					area = katsina_area
					area = zazzau_area
				}
			}
		}
		if = {
			limit = {
				not = { religion = sunni }
			}
			change_religion = sunni
		}
		cu_change_stability_point_n100 = yes	#Begining to purge the aristocrats.
		add_piety = 0.5
		set_country_flag = sok_dan_fodio_happened
	}
	option = {
		name = flavor_sok.1.b	#These dangerous ideas must be stopped.
		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				capital_scope = {
					OR = {
						area = hausa_area
						area = zazzau_area
						area = katsina_area
					}
				}
			}
		}
		cu_change_stability_point_n75 = yes
		add_piety = -0.1
		add_country_modifier = {
			name = ful_fulani_wars
			duration = 3650
		}
		custom_tooltip = flavor_sok.1.b.tt
		hidden_effect = {
			every_neighbor_country = {
				limit = {
					capital_scope = {
						OR = {
							area = hausa_area
							area = zazzau_area
							area = katsina_area
						}
					}
				}
				set_country_flag = sok_dan_fodio_happened
				country_event = { id = flavor_sok.2 }
			}
			every_known_country = {
				limit = {
					has_country_flag = fulani_jihad_state
					religion = sunni
					NOT = { has_country_flag = sok_dan_fodio_happened }
				}
				set_country_flag = sok_dan_fodio_happened
				country_event = { id = flavor_sok.2 }
			}
		}
	}
}
