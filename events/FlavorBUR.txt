					NOT = { has_country_flag = bur_marriage_eligible }
					NOT = { has_country_flag = bur_spain_eligible }

								is_part_of_hre = yes

									tag = CAS

										exists = no
									}
								}
								tag = SPA
							}
							num_of_cities = 4
							NOT = { tag = FRA }
						}
						set_country_flag = bur_marriage_successor
					}
				}
				if = {
					limit = {
						NOT = {
							any_country = {
								OR = {
									is_part_of_hre = yes
									AND = {
										tag = CAS
										SPA = {
											exists = no
