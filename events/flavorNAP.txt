# The Baronial Revolt
country_event = {
	id = flavor_nap.1
	title = flavor_nap.1.t
	desc = flavor_nap.1.d
	picture = ANGRY_MOB_eventPicture

	fire_only_once = yes

	trigger = {
		tag = NAP
		is_year = 1459
		NOT = { is_year = 1495 }
		NOT = { legitimacy = 90 }
		is_subject = no
		is_female = no
		is_lesser_in_union = no
		government = monarchy
		has_regency = no
		NOT = { dynasty = "de Valois" }
		exists = FRA
		FRA = {
			has_regency = no
			is_lesser_in_union = no
		}
	}
	
	mean_time_to_happen = {
		months = 36
	}
	
	option = {
		ai_chance = { factor = 90 }
		name = flavor_nap.1.a
		cu_change_stability_point_n75 = yes
		random_owned_province = {
			noble_rebels = 2
		}
		FRA = {
			add_opinion = {
				who = root
				modifier = nap_french_spurned
			}
		}
	}
	option = {
		name = flavor_nap.1.b
		FRA = {
			country_event = {
				id = flavor_nap.2
			}
			create_union = NAP
		}
		
	}
	
}

# Masaniello's Revolt
country_event = {
	id = flavor_nap.5
	title = flavor_nap.5.t
	desc = flavor_nap.5.d
	picture = ANGRY_MOB_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = NAP
		is_year = 1640
		NOT = { is_year = 1655 }
		OR = {
			NOT = { stability = 1 }
			overextension_percentage = 0.1
			war_exhaustion = 5
			inflation = 5
			NOT = { legitimacy = 90 }
			num_of_loans = 10
		}
	}
	mean_time_to_happen = {
		months = 1
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					NOT = { province_id = 121 }
				}
				set_province_flag = particularist_rebels
			}
			random_owned_province = {
				limit = {
					NOT = { province_id = 121 }
					NOT = { has_province_flag = particularist_rebels }
				}
				set_province_flag = particularist_rebels
			}
		}
	}

	option = {
		name = flavor_nap.5.a
		cu_change_stability_point_n150 = yes
		121 = {
			spawn_rebels = {
				type = particularist_rebels
				size = 2
				leader = "Masaniello"
			}
		}
		every_owned_province = {
			limit = {
				has_province_flag = particularist_rebels
			}
			particularist_rebels = 1
		}
	}
	option = {
		name = flavor_nap.5.b
		ai_chance = { factor = 90 }
		add_legitimacy = -25
		add_republican_tradition = -25
		add_corruption = 1
		121 = { particularist_rebels = 1 }

	}
}
