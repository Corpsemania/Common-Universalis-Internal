#Advancement of the Merchant classes
#Random +10/+20
country_event = {
	id = burghers_estate_events.1
	title = burghers_estate_events.1.t
	desc = burghers_estate_events.1.d
	picture = BURGHER_ESTATE_eventPicture
	
	trigger = {
		has_dlc = "The Cossacks"
		has_estate = estate_burghers
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_burghers
				modifier = EST_VAL_BURGHERS_ADVANCE
			}
		}
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_burghers
				modifier = EST_VAL_BURGHERS_ADVANCE_20
			}
		}
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_burghers
				modifier = EST_VAL_BURGHERS_DECLINE
			}
		}
	}

	is_triggered_only = yes
	
	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = burghers_estate_events.1.a #Ok
		add_estate_influence_modifier = {
			estate = estate_burghers
			desc = EST_VAL_BURGHERS_ADVANCE
			influence = 10
			duration = 5475
		}
	}
}

#Immoral Prices
#Force the Burghers to lose influence or face revolt in Burgher controlled provinces.
country_event = {
	id = burghers_estate_events.5
	title = burghers_estate_events.5.t
	desc = burghers_estate_events.5.d
	picture = BURGHER_ESTATE_eventPicture
	
	trigger = {
		has_dlc = "The Cossacks"
		has_estate = estate_burghers
		any_owned_province = {
			has_estate = estate_burghers
		}
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_burghers
				modifier = EST_VAL_FREE_OF_MEDDLING
			}
		}
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_burghers
				modifier = EST_VAL_PRICE_MEDDLING
			}
		}
	}
	
	is_triggered_only = yes
	
	mean_time_to_happen = {
		days = 1
		modifier = {
			factor = 2.0
			inflation = 5 #Rising Prices
		}
	} 
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					has_estate = estate_burghers
				}
				set_province_flag = bread_revolt_vs_burghers
			}
			random_owned_province = {
				limit = {
					has_estate = estate_burghers
					NOT = { has_province_flag = bread_revolt_vs_burghers }
					owner = { num_of_cities = 12 }
				}
				set_province_flag = bread_revolt_vs_burghers
			}
			random_owned_province = {
				limit = {
					has_estate = estate_burghers
					NOT = { has_province_flag = bread_revolt_vs_burghers }
					owner = { num_of_cities = 18 }
				}
				set_province_flag = bread_revolt_vs_burghers
			}
			random_owned_province = {
				limit = {
					has_estate = estate_burghers
					NOT = { has_province_flag = bread_revolt_vs_burghers }
					owner = { num_of_cities = 24 }
				}
				set_province_flag = bread_revolt_vs_burghers
			}
			random_owned_province = {
				limit = {
					has_estate = estate_burghers
					NOT = { has_province_flag = bread_revolt_vs_burghers }
					owner = { num_of_cities = 28 }
				}
				random = {
					chance = 75
					set_province_flag = bread_revolt_vs_burghers
				}
			}
			random_owned_province = {
				limit = {
					has_estate = estate_burghers
					NOT = { has_province_flag = bread_revolt_vs_burghers }
					owner = { num_of_cities = 32 }
				}
				random = {
					chance = 75
					set_province_flag = bread_revolt_vs_burghers
				}
			}
		}
	}
	
	option = {
		name = burghers_estate_events.5.a #Force them to sell at a just price.
		ai_chance = {
			factor = 75
			modifier = {
				factor = 0.2
				NOT = {
					estate_loyalty = {
						estate = estate_burghers
						loyalty = 45
					}
				}
			}
			modifier = {
				factor = 1.5
				estate_influence = {
					estate = estate_burghers
					influence = 65
				}
			}
		}
		hidden_effect = {
			every_owned_province = {
				limit = { has_province_flag = bread_revolt_vs_burghers }
				clr_province_flag = bread_revolt_vs_burghers
			}
		}
		add_estate_loyalty = {
			estate = estate_burghers
			loyalty = -20
		}
		add_estate_influence_modifier = {
			estate = estate_burghers
			desc = EST_VAL_PRICE_MEDDLING
			influence = -10
			duration = 5475
		}
	}
	option = {
		name = burghers_estate_events.5.b #We should not meddle with the Burghers.
		ai_chance = {
			factor = 25
			modifier = {
				factor = 2.0
				NOT = {

						estate = estate_burghers
						loyalty = 45
					}
				}
			}
			modifier = {
				factor = 0.5
				estate_influence = {
					estate = estate_burghers
					influence = 65
				}
			}

				size = 1
			}
		}
		add_estate_loyalty = {
