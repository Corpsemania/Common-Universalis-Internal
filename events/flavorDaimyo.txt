#Rise of the Chosokabe
country_event = {
	id = flavor_daimyo.1
	title = flavor_daimyo.1.t
	desc = flavor_daimyo.1.d
	picture = COURT_eventPicture
	
	fire_only_once = yes #No resurrections 

	trigger = {
		NOT = { is_year = 1560 }
		has_reform =  shogunate
		owns = 1819 #Tosa
		NOT = { tag = HSK }
		is_at_war = no
	}
	
	mean_time_to_happen = {
		months = 24
	}	
	
	option = {	
		name = flavor_daimyo.1.a #Let us empower the Chosokabe.
		ai_chance = {
			factor = 90
			modifier = {
				factor = 0
				NOT = { num_of_cities = 5 }
			}
			modifier = {
				factor = 0
				1819 = {
					is_capital = yes
				}
			}
		}
		add_legitimacy = 15
		add_yearly_manpower = 1.5
		1819 = {
			cede_province = CSK
			if = {
				limit = { NOT = { is_core = CSK } }
				add_core = CSK
			}
		}
		vassalize = CSK
		CSK = {
			add_opinion = {
				who = ROOT
				modifier = released_vassal
			}
			define_ruler = {
				name = "Motochika"
				dynasty = "Chosokabe"
				adm = 4
				dip = 4
				mil = 5
				fixed = yes
			}
		}
	}
	option = {	
		name = flavor_daimyo.1.b #We will keep it under our own control.
		ai_chance = {
			factor = 10
		}
		add_prestige = 5
		1819 = {
			if = {
				limit = { NOT = { is_core = CSK } }
				add_core = CSK
			}
			add_local_autonomy = 25
		}
	}
}

#Shiba Yoshikado
country_event = {
	id = flavor_daimyo.4
	title = flavor_daimyo.4.t
	desc = flavor_daimyo.4.d
	picture = BATTLE_eventPicture
	
	fire_only_once = yes
	is_triggered_only = yes #On new monarch

	trigger = {
		tag = SBA
		has_reform =  daimyo
		num_of_cities = 2
		OR = {
			has_regency = no
			has_heir = yes
			has_consort_regency = yes
		}
		NOT = { has_country_flag = shiba_yoshikado_flag } #Set in history
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					is_capital = no
					is_city = yes
				}
				save_event_target_as = yoshikado_pretender_province
			}
		}
	}
	
	option = {	
		name = flavor_daimyo.4.a #Stay true to [Root.Monarch.GetName]
		event_target:yoshikado_pretender_province = {
			spawn_rebels = {
				size = 1
				type = pretender_rebels
				leader = "Yoshikado Shiba"
			}
		}
	}
	option = {	
		name = flavor_daimyo.4.b #Support Yoshikado
		capital_scope = {
			if = {
				limit = { can_spawn_rebels = nationalist_rebels }
				spawn_rebels = {
					type = nationalist_rebels
					size = 1
				}
			}
			else = {
				spawn_rebels = {
					type = pretender_rebels
					size = 1
				}
			}
		}
	}
}
