	mean_time_to_happen = {

			limit = {
				manpower_percentage = 0.5

				add_unit_construction = {
					type = light_ship
					amount = 20
					speed = 0.1

				}
			}
		}

			add_opinion = {
				who = root
				modifier = mam_por_confronted

				id = flavor_mam.16
				days = 2

		}
		
		
	}
	option = {
		name = flavor_mam.15.b

			add_trade_modifier = {
