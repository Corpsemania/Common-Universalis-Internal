#Libertas Ecclesiae
#State making demands on the Church
#Christians
country_event = {
	id = church_estate_events.1
	title = church_estate_events.1.t
	desc = church_estate_events.1.d
	picture = CHURCH_ESTATE_1_eventPicture
	
	trigger = {
		has_dlc = "The Cossacks"
		has_estate = estate_church
		religion_group = christian
		estate_influence = {
			estate = estate_church
			influence = 25
		}
		NOT = {
			estate_influence = {
				estate = estate_church
				influence = 75
			}
		}
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_church
				modifier = EST_VAL_DEMANDED_CHURCH_MONEY
			}
		}
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_church
				modifier = EST_VAL_DIDNT_DEMAND_CHURCH_MONEY
			}
		}
	}
	
	is_triggered_only = yes
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = church_estate_events.1.a #Make it clear that they need to contribute.
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0.5
				NOT = {
					estate_loyalty = {
						estate = estate_church
						loyalty = 45
					}
				}
			}
			modifier = {
				factor = 0.5
				estate_influence = {
					estate = estate_church
					influence = 60
				}
			}
		}
		add_estate_loyalty = {
			estate = estate_church
			loyalty = -15
		}
		add_patriarch_authority = -0.1
		add_papal_influence = -10
		add_years_of_income = 0.35
		add_estate_influence_modifier = {
			estate = estate_church
			desc = EST_VAL_DEMANDED_CHURCH_MONEY
			influence = -10
			duration = 9125
		}
	}
	option = {
		name = church_estate_events.1.b #We will find the money elsewhere.
		add_prestige = 5
		add_piety = 0.1
		ai_chance = {
			factor = 50
			modifier = {
				factor = 1.5
				NOT = {
					estate_loyalty = {
						estate = estate_church
						loyalty = 40
					}
				}
			}
			modifier = {
				factor = 1.5
				estate_influence = {
					estate = estate_church
					influence = 60
				}
			}
		}
		add_estate_loyalty = {
			estate = estate_church
			loyalty = 10
		}
		add_estate_influence_modifier = {
			estate = estate_church
			desc = EST_VAL_DIDNT_DEMAND_CHURCH_MONEY
			influence = 10
			duration = 9125
		}
	}
}

country_event = {
	id = church_estate_events.4
	title = church_estate_events.4.t
	desc = church_estate_events.4.d
	picture = CHURCH_ESTATE_1_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		has_dlc = "The Cossacks"
		has_estate = estate_church
		has_estate = estate_burghers
		any_owned_province = {
			has_estate = estate_burghers
		}
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_church
				modifier = EST_VAL_TITHES_DEFENDED
			}
		}
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_church
				modifier = EST_VAL_TITHES_UNDERMINED
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = church_estate_events.4.a #Then we must force them to pay!
		ai_chance = {
			factor = 50
			modifier = {
				factor = 1.5
				NOT = {
					estate_loyalty = {
						estate = estate_church
						loyalty = 40
					}
				}
			}
			modifier = {
				factor = 0.5
				estate_influence = {
					estate = estate_church
					influence = 60
				}
			}
			modifier = {
				factor = 0.5
				NOT = {
					estate_loyalty = {
						estate = estate_burghers
						loyalty = 40
					}
				}
			}
			modifier = {
				factor = 1.5
				estate_influence = {
					estate = estate_burghers
					influence = 60
				}
			}
		}
		add_years_of_income = 0.2
		random_owned_province = {
			limit = {
				has_estate = estate_burghers
				local_autonomy = 25
			}
			add_local_autonomy = -25 #If a province with Burghers have more than 25 it will be reduced by this much or down to 25.
			hidden_effect = { change_variable = { which = epi value = -2 } }
		}
		add_estate_loyalty = {
			estate = estate_church
			loyalty = 15
		}
		add_estate_loyalty = {
			estate = estate_burghers
			loyalty = -15
		}
		add_estate_influence_modifier = {
			estate = estate_church
			desc = EST_VAL_TITHES_DEFENDED
			influence = 10
			duration = 5475
		}
		add_estate_influence_modifier = {
			estate = estate_burghers
			desc = EST_VAL_TITHES_DEFENDED
			influence = -10
			duration = 5475
		}
	}
	option = {
		name = church_estate_events.4.b #A mild reprimand will be enough.
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0.5
				NOT = {
					estate_loyalty = {
						estate = estate_church
						loyalty = 40
					}
				}
			}
