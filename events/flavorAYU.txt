# Prasat T'ang's Economic Reforms
country_event = {
	id = flavor_ayu.3
	title = "flavor_ayu.EVTNAME3"
	desc = "flavor_ayu.EVTDESC3"
	picture = REFORM_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = AYU
		is_year = 1600
		NOT = { is_year = 1700 }
	}
	
	mean_time_to_happen = {
		months = 200
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					is_core = AYU
				}
				set_province_flag = AYU_eco_reform_flag1
			}
			random_owned_province = {
				limit = {
					NOT = {
						has_province_flag = AYU_eco_reform_flag1
					}
				}
				set_province_flag = AYU_eco_reform_flag2
			}
			random_owned_province = {
				limit = {
					NOT = {
						has_province_flag = AYU_eco_reform_flag1
					}
					NOT = {
						has_province_flag = AYU_eco_reform_flag2
					}
				}
				set_province_flag = AYU_eco_reform_flag3
			}
		}
	}
	
	option = {
		name = "flavor_ayu.EVTOPTA3" #Go with the reforms
		cu_change_stability_point_n75 = yes
		add_mercantilism = 2
		add_inflation = -5
		random_owned_province = {
			limit = {
				has_province_flag = AYU_eco_reform_flag1
			}
			add_base_tax = 1
			clr_province_flag = AYU_eco_reform_flag1
		}
		random_owned_province = {
			limit = {
				has_province_flag = AYU_eco_reform_flag2
			}
			add_base_tax = 1
			clr_province_flag = AYU_eco_reform_flag2
		}
		random_owned_province = {
			limit = {
				has_province_flag = AYU_eco_reform_flag3
			}
			add_base_tax = 1
			clr_province_flag = AYU_eco_reform_flag3
		}
	}
	option = {
		name = "flavor_ayu.EVTOPTB3" #Ignore the reform ideas
		cu_change_stability_point_p50 = yes
	}
}

	desc = "flavor_ayu.EVTDESC6"
	picture = BUDDHA_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = AYU
		is_year = 1625
		NOT = { is_year = 1675 }
		has_regency = no
		OR = {
			dip = 4
			mil = 4
			adm = 4
		}
		NOT = { has_ruler_modifier = ayu_wat_ratchaburan_temple }
	}
	
	mean_time_to_happen = {
		months = 200
	}
	
	option = {
		name = "flavor_ayu.EVTOPTA6" #Sponsor the temple
		add_years_of_income = -0.2
		add_ruler_modifier = {
			name = ayu_wat_chaiwatthanaram_temple
		}
		hidden_effect = {
   		 set_variable = { which = cu_var1 value = 3  }
		}
		cu_change_cultural_legacy = yes
		custom_tooltip = cu_change_cultural_legacy_3t
	}
	option = {
		name = "flavor_ayu.EVTOPTB6" #Send help
		add_adm_power = -20
		add_prestige = 10

   		 set_variable = { which = cu_var1 value = 2  }

		custom_tooltip = cu_change_cultural_legacy_2t
	}
}
