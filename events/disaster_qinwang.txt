country_event = {
	id = disaster_qinwang.4
	title = disaster_qinwang.4.t
	desc = disaster_qinwang.4.d
	picture = qinwang_ESTATE_1_eventPicture
	hidden = yes
	mean_time_to_happen = {
		months = 3
	}
	
	trigger = {
		has_disaster = estate_qinwang_disaster
		NOT = { has_country_flag = other_estate_joined }
	}
	immediate = {
		set_country_flag = other_estate_joined
	}
	option = {
		name = OK
		if = {
			limit = { has_estate = estate_shizu }
			if = {#loyal?
				limit = {#loyalty&Reforms
					OR = {
						estate_loyalty = {
							estate = estate_shizu
							loyalty = 45
						}
						AND = {# protect (or gain?) privileges
							estate_loyalty = {
								estate = estate_shizu
								loyalty = 40
							}
							OR = {
								has_reform = clergy_reform
								has_reform = divine_law
								has_reform = tax_policy_4
								has_reform = tax_policy_5
								has_reform = cu_religious_policy_4
								has_reform = cu_religious_policy_5
							}
						}
					}
				}
				country_event = { id = disaster_shishen.10 }#Alliance ? Reject ? Or just ask for help?
			}
			else_if = {#disloyal?
				limit = {
					OR = {
						AND = {
							OR = {
								has_reform = court_despotism
								has_reform = sovereign_code
								has_reform = no_election
								has_reform = tax_policy_1
								has_reform = tax_policy_2
								has_reform = cu_religious_policy_1
								NOT = {
									estate_loyalty = {
										estate = estate_shizu
										loyalty = 30
									}
								}
							}
							NOT = {
								estate_loyalty = {
									estate = estate_shizu
									loyalty = 35
								}
							}
						}
						has_reform = cu_religious_policy_2
					}
				}
				country_event = { id = disaster_shishen.9 }#They joined revolt
			}
		}
		
		if = {
			limit = { has_estate = estate_burghers }
			if = {#loyal?
				limit = {#loyalty&Reforms
					OR = {
						estate_loyalty = {
							estate = estate_burghers
							loyalty = 60
						}
						AND = {# protect (or gain?) privileges
							estate_loyalty = {
								estate = estate_burghers
								loyalty = 40
							}
							OR = {
								has_reform = burghers_election
								has_reform = universal_suffrage
								has_reform = tax_policy_1
								has_reform = tax_policy_2
								has_reform = education_policy_burghers
								has_reform = education_policy_public
								has_reform = parliament_reform
							}
						}
					}
				}
				country_event = { id = disaster_burghers.10 }#Alliance ? Reject ? Or just ask for help?
			}
			else_if = {#disloyal?
				limit = {
					OR = {
						has_reform = court_despotism
						has_reform = sovereign_code
						has_reform = no_election
						has_reform = elective_monarchy
						has_reform = tax_policy_4
						has_reform = tax_policy_5
						has_reform = noble_council
						has_reform = clergy_reform
						has_reform = divine_law
						has_reform = noble_election
						NOT = {
							estate_loyalty = {
								estate = estate_burghers
								loyalty = 30
							}
						}
					}
					NOT = {
						estate_loyalty = {

						}

				}
