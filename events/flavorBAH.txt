#Bidar, a true capital for the Bahmanis
country_event = {
	id = flavor_bah.1
	title = flavor_bah.1.t
	desc = flavor_bah.1.d
	picture = COURT_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = BAH
		religion_group = muslim
		has_ruler = "Ala ud-Din Ahmad Shah II"
		years_of_income = 0.25
	}
	
	mean_time_to_happen = {
		months = 18
	}

	option = {
		name = flavor_bah.1.a #Bidar must remain a capital worth of our dynasty.
		add_adm_power = -50
		add_years_of_income = -0.2
		capital_scope = {
			add_province_modifier = {
				name = "bah_bidar_fort"
				duration = -1
			}
		}
	}
	
	option = {
		name = flavor_bah.1.b #Let us not go overboard with this.
		add_prestige = -5
	}
}

#Mahmud Gawan
country_event = {
	id = flavor_bah.3
	title = flavor_bah.3.t
	desc = flavor_bah.3.d
	picture = ADVISOR_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = BAH
		religion_group = muslim
		is_year = 1450
		NOT = { is_year = 1480 }
		OR = {
			prestige = 80
			has_country_flag = bah_pr_campaign
		}
	}
	
	mean_time_to_happen = {
		months = 90
	}

	option = {
		name = flavor_bah.3.a
		add_adm_power = 25
		add_dip_power = 25
		add_meritocracy_effect = yes
		define_advisor = {
			type = trader
			name = "Mahmud Gawan"
			culture = persian
			religion = shiite
			skill = 3
			discount = yes
		}
		hidden_effect = {
   		 set_variable = { which = cu_var1 value = 1  }
		}
		cu_change_cultural_legacy = yes
		custom_tooltip = cu_change_cultural_legacy_1t
	}
}
