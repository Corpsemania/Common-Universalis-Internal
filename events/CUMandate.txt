		}

	desc = "tribute.30.d"
	picture = COURT_eventPicture
	
        trigger = {
                   has_country_flag = send_tribute
        }

        mean_time_to_happen = {
                months = 6
        }
         

	option = { #OK
		name = "tribute.30.a" 
                random_known_country = {
                        limit = { has_opinion_modifier = { who = ROOT modifier = opinion_tribute_target } }

                                limit = { ROOT = { has_country_flag = tribute_money } }

                        }
                        else_if = {
                                limit = { ROOT = { has_country_flag = tribute_student } }
								add_opinion = { who = ROOT modifier = opinion_tribute_student }
                                country_event = { id = mandate.8021}
                        }
                        else_if = {
                                limit = { ROOT = { has_country_flag = tribute_empty } }
								add_opinion = { who = ROOT modifier = opinion_tribute_empty }
                                country_event = { id = mandate.8022}
                        }
                        hidden_effect = { remove_opinion = { who = ROOT modifier = opinion_tribute_target } }
                }
				clr_country_flag = tribute_money
				clr_country_flag = tribute_student
				clr_country_flag = tribute_empty
        }
}
