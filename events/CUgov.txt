province_event = {
	id = govstart.253 #Core Effect
	title = "govstart_titl"
	desc = "govstart_desc"
	picture = COLONIZATION_eventPicture
	is_triggered_only = yes
	hidden = yes
	
	
	option = {
		name = "govstart.a"
		if = {
		    limit = { 
			    OR = { #Reclaimed lost territory, transferred province due to forming/releasing new nation, integrated vassal etc. starts with max integrity
				    is_core = owner
					is_territorial_core = owner
					has_province_flag = cu_colonial_transfer
				}
			}
			clr_province_flag = cu_colonial_transfer
			cu_calc_base_cultural_acceptance = yes
			set_variable = { which = p_ca which = p_cab }
			if = { #Colony set to best condition when transfered to colonial nation
				limit = { has_province_flag = cu_colonial_transfer }
				set_local_autonomy = 20
				set_variable = { which = p_ca value = 100 }
				cu_check_cultural_acceptance = yes
			}
		}
		else_if = {
		        limit = { is_colony = yes } #Colonial province starts with 70% of max integrity
				cu_calc_base_cultural_acceptance = yes
				set_variable = { which = p_ca which = p_cab }
				multiply_variable = { which = p_ca value = 0.7 }
				set_province_flag = cu_colony_flag
		}
		else_if = {
				limit = {
					has_owner_culture = yes #Same culture province starts with 50% of max integrity
				}
				set_province_flag = cu_newly_conquered_province
				cu_calc_base_cultural_acceptance = yes
				set_variable = { which = p_ca which = p_cab }
				multiply_variable = { which = p_ca value = 0.5 }
		}
		else_if = {
		        limit = {
				       owner = { culture_group = ROOT }  #Culture from same group start with 40% of max integrity
				}
				set_province_flag = cu_newly_conquered_province
				cu_calc_base_cultural_acceptance = yes
				set_variable = { which = p_ca which = p_cab }
				multiply_variable = { which = p_ca value = 0.4 }
		}
		else_if = {
		        limit = {
				       religion = owner  #Same religion start with 30% of max integrity
				}
				set_province_flag = cu_newly_conquered_province
				cu_calc_base_cultural_acceptance = yes
				set_variable = { which = p_ca which = p_cab }
				multiply_variable = { which = p_ca value = 0.3 }
		}
		else_if = {
		       limit = {
				       religion_group = owner  #Same religion group start with 20% of max integrity
				}
				set_province_flag = cu_newly_conquered_province
				cu_calc_base_cultural_acceptance = yes
				set_variable = { which = p_ca which = p_cab }
				multiply_variable = { which = p_ca value = 0.2 }
		}
		else = { #Start from ash
		        set_province_flag = cu_newly_conquered_province
				cu_calc_base_cultural_acceptance = yes
				set_variable = { which = p_ca value = 0 }
		}
		
		#Add province core
		if = { #HRE Province waits two months before coring, so that the emperor has time to demand its return.
			limit = { is_part_of_hre = yes }
			province_event = { id = core.6 days = 180 }
		}
		else_if = { #Conquered colonial provinces, trigger slaughtering native effect
		    limit = { 
				has_province_flag = cu_newly_conquered_province
				religion_group = pagan
				owner = { 
					OR = {
						religion_group = christian 
						religion_group = muslim
						has_reform = cu_religious_policy_5
					}
					#NOT = { has_reform = cu_religious_policy_1 }
					#NOT = { has_reform = cu_religious_policy_2 }
				}
				OR = { 
						continent = north_america
						continent = south_america
						continent = new_world
				}
				NOT = { culture = owner }
				owner = { mil_power = 0 }

			}
			set_province_flag = cu_wait_slaughter
			add_core = owner
		}
		else_if = {
		    limit = { #Mandate of Heaven mechanic
				OR = {
					superregion = china_superregion
					culture_group = east_asian
				}
				AND = {
						owner = {
							capital_scope = { 
									OR = {
										region = south_china_region
										region = xinan_region
										region = north_china_region
									} 
							}	
							OR = {
								has_reform =  chinese_revolter
								has_reform =  zhuhou
								has_reform =  celestial_empire
								has_reform =  celestial_empire_2
								has_reform =  celestial_empire_3
								
							}
						}
				}
			}
			add_core = owner
			if = {
			    limit = { NOT = { check_variable = { which = p_ca value = 30 } } }
				set_variable = { which = p_ca value = 30 }
			}
			else = { 
				change_variable = { which = p_ca value = 10 }
			}
		}
		else_if = {
			limit = {
			       is_permanent_claim = owner
			}
			add_core = owner
			change_variable = { which = p_ca value = 10 }
		}
		else_if = {
			limit = {
			       is_claim = owner
			}
			add_core = owner
			change_variable = { which = p_ca value = 5 }
		}
		else = {
			add_core = owner
		}
		
		cu_check_cultural_acceptance = yes
	}
}
