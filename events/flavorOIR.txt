# Yingzong Emperor leads the army against the Oirats
country_event = {
	id = flavor_oir.1
	title = flavor_oir.1.t
	desc = flavor_oir.1.d
	picture = MONGOL_BANNERS_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = MNG
		war_with = OIR
		NOT = { is_year = 1500 }
		NOT = { MIL = 2 }
		NOT = { has_regency = yes }
		is_female = no
	}
	
	mean_time_to_happen = {
		months = 1
	}
	
	option = { # what an idiot
		name = "flavor_oir.1.a"
		if = {
			limit = {
				NOT = { is_monarch_leader = yes }
			}
			define_ruler_to_general = {
				fire = 1
				shock = 0
				manuever = 0
				siege = 0
			}
		}
		set_ruler_flag = mng_militry_incompetence
	}
}

	is_triggered_only = yes

			limit = {
				is_emperor_of_china = yes
			}
			add_mandate = -20
		}
		cu_change_stability_point_n75 = yes
		hidden_effect = {
			if = {
				limit = {
					has_heir = yes
				}
				exile_heir_as = mng_exiled_heir
				set_country_flag = mng_exiled_heir_flag
			}
			exile_ruler_as = mng_exiled_ruler
		}
		define_ruler = {
			regency = yes
			ADM = 2
			DIP = 2
			MIL = 2
			fixed = yes
		}
	}
}
	
# Ming appoints new Emperor
country_event = {
	id = flavor_oir.4
	title = flavor_oir.4.t
	desc = {
		trigger = {
			has_heir = yes
		}
		desc = flavor_oir.4.d
	}
	desc = {
		trigger = {
			has_heir = no
		}
		desc = flavor_oir.4.d2
	}
	picture = BAD_WITH_MONARCH_eventPicture
	
	fire_only_once = yes
	major = yes
	
	trigger = {
		tag = MNG
		war_with = OIR
		had_country_flag = {
			flag = mng_emperor_captured
			days = 1825
		}
		NOT = { has_country_flag = oir_demands_sent }
		NOT = {
			1816 = {
				num_of_units_in_province = {
					who = OIR
					amount = 9
				}
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			OIR = {
				clr_country_flag = captured_mng_emperor
			}
			clr_country_flag = mng_emperor_captured
		}
	}
	
	mean_time_to_happen = {
		months = 1
	}
	
	option = { # May he have better judgement than his predecessor
		name = "flavor_oir.4.a"
		if = {
			limit = {
				has_country_flag = mng_exiled_heir_flag
			}
			set_heir = mng_exiled_heir
		}
		else = {
			cu_change_stability_point_n75 = yes
			define_ruler = {
				dynasty = "Zhu"
				culture = ROOT
				religion = ROOT
				male = yes
			}
		}
	}
}

			MNG = {
				every_owned_province = {
					limit = {
						region = north_china_region

					}
					change_controller = OIR
				}
			}
		}
	}
}

# The Dzunghars
country_event = {
	id = flavor_oir.8
	title = flavor_oir.8.t
	desc = flavor_oir.8.d
	picture = MONGOL_BANNERS_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		is_year = 1600
		NOT = { exists = ZUN }
		any_owned_province = {
			area = north_zungaria
		}
		OR = {
			NOT = { exists = OIR }
			NOT = {
				OIR = {
					num_of_cities = 10
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 12
		modifier = {
			factor = 0.5
			num_of_owned_provinces_with = {
				OR = {
					area = north_zungaria
					area = zungaria_area
				}
				value = 5
			}
		}
	}
	
	option = {
		name = "flavor_oir.8.a"
		north_zungaria = {
			add_core = ZUN
		}
		zungaria_area = {
			add_core = ZUN
		}
	}
}
