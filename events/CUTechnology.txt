namespace = tech

country_event = {#reclac adm tech cost modifier
	id = tech.1
	title = "tech.1.t"
	desc = "tech.1.d"
	picture = ADVISOR_eventPicture
	hidden = yes
	is_triggered_only = yes
	trigger = {
		OR = {
			NOT = { has_country_flag = at_check }
			AND = {
				ai = no
				had_country_flag = {
					flag = at_check
					days = 31
				}
			}
			had_country_flag = {
				flag = at_check
				days = 730
			}
		}
	}
	#mean_time_to_happen = {
		#days = 1
	#}
	option = {
		name = "tech.1.a"
		clr_country_flag = at_check
		set_country_flag = at_check
		set_variable = { which = tla value = 0 }
		export_to_variable = {
			which = tla
			value = adm_tech
		}
		if = {
			limit = {
				check_variable = {
					which = tla
					which = ntla
				}
			}
			set_variable = { which = resa value = 0 }
			cu_calc_adm_tech_cost = yes
		}
		else = {
		     cu_finish_adm_tech = yes
		}
	}
}
