# The Meeting at Salsipuedes
country_event = {
	id = flavor_cua.1
	title = flavor_cua.1.t
	desc = flavor_cua.1.d
	picture = DIPLOMACY_eventPicture

	fire_only_once = yes
	
	trigger = {
		tag = CUA
		any_neighbor_country = {
			capital_scope = { continent = europe }
			truce_with = ROOT
		}
	}
	
	mean_time_to_happen = {
		months = 32
	}
	
	option = {
		name = flavor_cua.1.a
		random_neighbor_country = {
			limit = {
				capital_scope = { continent = europe }
				truce_with = ROOT
			}
			add_opinion = { who = ROOT modifier = cua_refused_meeting }
		}
	}
	option = {
		name = flavor_cua.1.b
		country_event = {
			id = flavor_cua.2
			days = 7
			tooltip = flavor_cua.1.tt
		}
	}
}

# The Salsipuedes Massacre
country_event = {
	id = flavor_cua.2
	title = flavor_cua.2.t
	desc = flavor_cua.2.d
	picture = BATTLE_eventPicture

	is_triggered_only = yes
	
	option = {
		name = flavor_cua.2.a
		add_manpower = -2
		random_neighbor_country = {
			limit = {
				has_opinion_modifier = { who = ROOT modifier = cua_refused_meeting }
			}
			reverse_add_opinion = { who = ROOT modifier = cua_salsipuedes_massacre }
			reverse_add_casus_belli = {
				target = ROOT
				type = cb_border_war
				months = 24
			}
		}
	}
}

# Juan Díaz de Solís
country_event = {
	id = flavor_cua.3
	title = flavor_cua.3.t
	desc = flavor_cua.3.d
	picture = EXPLORERS_eventPicture

	fire_only_once = yes
	
	trigger = {
		tag = CUA
		any_known_country = {
			knows_country = ROOT
			capital_scope = { continent = europe }
			any_owned_province = {
				region = la_plata_region
			}
		}
	}
	
	mean_time_to_happen = {
		months = 70
	}
	
	option = {
		name = flavor_cua.3.a
		add_stability_or_adm_power = yes
		random_known_country = {
			limit = {
				knows_country = ROOT
				capital_scope = { continent = europe }
				any_owned_province = {
					region = la_plata_region
				}
			}
			add_opinion = { who = ROOT modifier = cua_killed_juan }
		}
	}
	option = {
		name = flavor_cua.3.b
		random_owned_province = {
			add_province_modifier = {
				name = add_unrest_5_modifier
				duration = 1825
			}
		}
		random_known_country = {
			limit = {
				knows_country = ROOT
				capital_scope = { continent = europe }
				any_owned_province = {
					region = la_plata_region
				}
			}
			add_opinion = { who = ROOT modifier = cua_released_juan }
		}
	}
	option = {
		name = flavor_cua.3.c
		trigger = {
			primitives = no
		}
		add_prestige = -5
		random_owned_province = {
			add_province_modifier = {
				name = add_unrest_5_modifier
				duration = 1825
			}
		}
		define_explorer = {
			name = "Juan Díaz de Solís"
			fire = 2
			shock = 3
			manuever = 2
			siege = 1
		}
	}
}
