# Onin War
country_event = {
	id = flavor_jap.1
	title = "flavor_jap.EVTNAME1"
	desc = "flavor_jap.EVTDESC1"
	picture = BATTLE_eventPicture

	fire_only_once = yes
	
	trigger = {
		has_reform =  daimyo
		any_country = {
			offensive_war_with = ROOT
			has_reform =  daimyo
			is_in_war = {
				defender_leader = ROOT
				attacker_leader = THIS
			}
		}
	}

	immediate = {
		hidden_effect = {
			save_event_target_as = onin_defensive_war_leader
			random_country = {
				limit = {
					offensive_war_with = ROOT
					has_reform =  daimyo
					is_in_war = {
						defender_leader = ROOT
						attacker_leader = THIS
					}
				}
				save_event_target_as = onin_offensive_war_leader
			}
			every_country = {
				limit = {
					NOT = { tag = ROOT }
					OR = {
						has_reform =  daimyo
						has_reform =  shogunate
					}
				}
				country_event = { id = flavor_jap.56 }
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "flavor_jap.EVTOPTA1"
		custom_tooltip = shogunate_mechanic_explanation
		add_country_modifier = {
			name = jap_sengoku_jidai
			duration = 10950
		}
	}
}
