country_event = {
	id = flavor_gua.1
	title = flavor_gua.1.t
	desc = flavor_gua.1.d
	picture = BATTLE_eventPicture

	trigger = {
		tag = GUA
		any_neighbor_country = {
			OR = {
				capital_scope = {
					continent = europe
				}
				colonial_parent = {
					capital_scope = {
						continent = europe
					}
				}
			}
			knows_country = ROOT
			ROOT = {
				knows_country = PREV
			}
		}
	}
	mean_time_to_happen = {
		months = 32
	}
	
	option = {
		name = flavor_gua.1.a
		cu_change_stability_point_n75 = yes
		random_neighbor_country = {
			limit = {
				capital_scope = { continent = europe }
			}
			country_event = {
				id = flavor_gua.2
				days = 7
				tooltip = flavor_gua.2.tt
			}
		}
	}
	option = {
		name = flavor_gua.1.b
		cu_change_stability_point_p50 = yes
		random_neighbor_country = {
			limit = {
				capital_scope = { continent = europe }

				ROOT = {
					knows_country = PREV
				}
			}
			add_opinion = { who = ROOT modifier = gua_oppose_bandeirantes }
		}
	}
}

country_event = {
	id = flavor_gua.4
	title = flavor_gua.4.t
	desc = flavor_gua.4.d
	picture = TRADEGOODS_eventPicture

	#fire_only_once = yes
	
	trigger = {
		tag = GUA
		is_at_war = no
		has_regency = no
		NOT = {
			has_ruler_modifier = gua_yerba_mate
		}
		any_owned_province = {
			trade_goods = tea
		}
	}
	
	mean_time_to_happen = {
		months = 120
	}
	
	option = {
		name = flavor_gua.4.a
		add_adm_power = -10
		add_ruler_modifier = {
			name = gua_yerba_mate
		}
		hidden_effect = {
   		 set_variable = { which = cu_var1 value = 1 }
		}
		cu_change_cultural_legacy = yes
		custom_tooltip = cu_change_cultural_legacy_1t
	}
	option = {
		name = flavor_gua.4.b
		add_prestige = -5
		random_owned_province = {
			limit = {
				trade_goods = tea
			}
			add_province_modifier = {
				name = gua_yerba_mate_production
				duration = 1825
			}
		}
	}
}
