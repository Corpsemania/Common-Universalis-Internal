#Nobles Demand Exlusive right to Commissions
country_event = {
	id = nobles_estate_events.1
	title = nobles_estate_events.1.t
	desc = nobles_estate_events.1.d
	picture = NOBLE_ESTATE_eventPicture
	
	trigger = {
		has_dlc = "The Cossacks"
		has_estate = estate_nobles
		estate_influence = {
			estate = estate_nobles
			influence = 70
		}
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_nobles
				modifier = EST_VAL_EXCLUSIVE_RIGHT_TO_BE_OFFICERS_NOBLES
			}
		}
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_nobles
				modifier = EST_VAL_NO_EXCLUSIVE_RIGHT_TO_BE_OFFICERS_NOBLES
			}
		}
	}
	
	is_triggered_only = yes
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = nobles_estate_events.1.a #
		ai_chance = {
			factor = 50
			modifier = {
				factor = 1.5
				NOT = {
					estate_loyalty = {
						estate = estate_nobles
						loyalty = 40
					}
				}
			}
			modifier = {
				factor = 1.2
				estate_loyalty = {
					estate = estate_nobles
					loyalty = 50
				}
				NOT = {
					estate_loyalty = {
						estate = estate_nobles
						loyalty = 60
					}
				}
			}
			modifier = {
				factor = 0.2
				NOT = {
					estate_loyalty = {
						estate = estate_burghers
						loyalty = 30
					}
				}
			}
		}
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = 15
		}
		add_estate_loyalty = {
			estate = estate_burghers
			loyalty = -20
		}
		add_country_modifier = {
			name = "only_noble_officers"
			duration = 7300
		}
		add_estate_influence_modifier = {
			estate = estate_nobles
			desc = EST_VAL_EXCLUSIVE_RIGHT_TO_BE_OFFICERS_NOBLES
			influence = 10
			duration = 7300
		}
	}
	option = {
		name = nobles_estate_events.1.b #
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0.1
				NOT = {
					estate_loyalty = {
						estate = estate_burghers
						loyalty = 40
					}
				}
			}
			modifier = {
				factor = 1.5
				NOT = {
					estate_loyalty = {
						estate = estate_nobles
						loyalty = 30
					}
				}
			}
		}
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = -20
		}
		add_estate_loyalty = {
			estate = estate_burghers
			loyalty = 15
		}
		add_army_professionalism = 0.02
		add_estate_influence_modifier = {
			estate = estate_nobles
			desc = EST_VAL_NO_EXCLUSIVE_RIGHT_TO_BE_OFFICERS_NOBLES
			influence = -10
			duration = 7300
		}
	}

#Nobles encroach on Government Lands
country_event = {
	id = nobles_estate_events.3
	title = nobles_estate_events.3.t
	desc = nobles_estate_events.3.d
	picture = NOBLE_ESTATE_eventPicture
	
	trigger = {
		has_dlc = "The Cossacks"
		has_estate = estate_nobles
		num_of_cities = 15
		any_owned_province = {
			NOT = { has_estate = yes }
			is_overseas = no
			is_capital = no
			is_colony = no
			has_seat_in_parliament = no
			cu_not_terr_trigger = yes
		}
	}
	
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					NOT = { has_estate = yes }
					is_capital = no
					is_overseas = no
					is_colony = no
					has_seat_in_parliament = no
					cu_not_terr_trigger = yes
				}
				set_province_flag = noble_encroachment
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
		modifier = {
			factor = 2.0
			estate_influence = {
				estate = estate_nobles
				influence = 80
			}
			NOT = {
				estate_loyalty = {
					estate = estate_nobles
					loyalty = 80
				}
			}
		}
	}
	
	option = {
		name = nobles_estate_events.3.a #Allow it.
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0.25
				estate_territory = {
					estate = estate_nobles
					territory = 0.12
				}
			}
			modifier = {
				factor = 0.25
				estate_influence = {
					estate = estate_nobles
					influence = 60
				}
			}
			modifier = {
				factor = 1.5
				NOT = {
					estate_loyalty = {
						estate = estate_nobles
						loyalty = 30
					}
				}
			}
		}
		random_owned_province = {
			limit = {
				has_province_flag = noble_encroachment
			}
			set_estate = estate_nobles
			clr_province_flag = noble_encroachment
			add_province_modifier = {
				name = new_estate_rise_n #Estate
				duration = 1825
			}
		}
	}
	option = {
		name = nobles_estate_events.3.b #Take it Back.
		ai_chance = {
			factor = 50
			modifier = {
				factor = 1.1
				NOT = {
					estate_territory = {
						estate = estate_nobles
						territory = 0.15
					}
				}
			}
			modifier = {
				factor = 0.25
				NOT = {
					estate_loyalty = {
						estate = estate_nobles
						loyalty = 35
					}
				}
			}
		}
		random_owned_province = {
			limit = {
				has_province_flag = noble_encroachment
			}
			random_list = {
				50 = {
					spawn_rebels = {
						type = noble_rebels
						size = 1
					}
				}
				50 = {  }
			}
			clr_province_flag = noble_encroachment
			change_variable = { which = epi value = 5 }
		}
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = -15
		}
	}
}

		}
	}

	mean_time_to_happen = {
		days = 1
	}
	
	option = { #Indifferent choice, likely appropriate if estate is just right.

			factor = 33
