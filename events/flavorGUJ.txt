# Growth of Ahmedabad
country_event = {
	id = flavor_guj.1
	title = flavor_guj.1.t
	desc = flavor_guj.1.d
	picture = CITY_VIEW_eventPicture
	fire_only_once = yes

	trigger = {
		tag = GUJ
		NOT = { is_year = 1475 }
		capital = 2054 #Ahmedabad
		NOT = { has_dlc = "Dharma" } #Integrated into Dharma Missions													   
	}

	mean_time_to_happen = {
		months = 90
	}
	
	
	option = { 
		name = flavor_guj.1.a
		ai_chance = { factor = 95 }
		add_years_of_income = -0.1
		add_adm_power = -50
		2054 = {
			add_province_modifier = {
				name = guj_walls_of_ahmedabad
				duration = -1
			}
		}
	}
	option = {
		name = flavor_guj.1.b	
		ai_chance = { factor = 5 }
	}
	
}

				owns_core_province = 2315

				owns_core_province = 2321
			}
		}
		 POR = {
			any_owned_province = {
				superregion = india_superregion
			}
			}
	}

		months = 42
	}
	
	
	option = {
		name = flavor_guj.7.a #Contact the Sultan

		TUR = { country_event = { id = flavor_guj.8 days = 30 } }
	}
	
	option = {
		name = flavor_guj.7.b #We will manage fine on our own.
		ai_chance = { factor = 10 }
	}
}
