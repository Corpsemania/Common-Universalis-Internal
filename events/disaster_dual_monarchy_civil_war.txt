country_event = {   
    id = dualmonarchycivilwar.1
	title = "dualmonarchycivilwar.1.t"
	desc = "dualmonarchycivilwar.1.d"
	picture = ECONOMY_eventPicture
	
	is_triggered_only = yes
	immediate = {
		hidden_effect = {
			every_owned_province = {
				limit = { 
					OR = {
						culture = english
						culture = scottish
					}
					continent = europe						  
				}
				add_nationalism = 15
			}
			set_variable = { which = cu_var1 value = -100 }
		}
		
		
	}
	option = {
		name = "dua.1.a"	
		hidden_effect = {
			set_variable = { which = cu_var1 value = -100 }
		}
		cu_change_stability_point = yes		
		if = {
          limit = { dip = 3 }
		  random_list = {
			30 = { country_event = {  id = dualmonarchycivilwar.49 } }
			10 = { country_event = {  id = dualmonarchycivilwar.50 } }
			60 = { country_event = {  id =  dualmonarchycivilwar.2 } }
		  }
		}
		if = {
          limit = { NOT = { dip = 3 } }
		  random_list = {
			35 = { country_event = {  id = dualmonarchycivilwar.49 } }
			25 = { country_event = {  id = dualmonarchycivilwar.50 } }
			40 = { country_event = {  id =  dualmonarchycivilwar.2 } }
		  }
		}
	}
	option = {
		name = "dua.1.b" 
		hidden_effect = {
			set_variable = { which = cu_var1 value = -100 }
		}
		cu_change_stability_point = yes		
		hidden_effect = {
		every_owned_province = {
		       limit = { 
			            OR = {
							culture = english
							culture = scottish
						}
                        continent = europe						  
			   }
				add_province_modifier = {
				name = suppress_minority
				duration = 3650
			}
		}
		random_owned_province = {
			limit = {
				culture = english
				continent = europe	
			}
			spawn_rebels = {
				type = nationalist_rebels
				size = 3
				friend = ENG
			}
		}
		random_owned_province = {
			limit = {
				culture = scottish
				continent = europe	
			}
			spawn_rebels = {
				type = nationalist_rebels
				size = 3
				friend = SCO
			}
		}

		}
