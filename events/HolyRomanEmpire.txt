	id = 9075
	title = "EVTNAME9075"
	desc = "EVTDESC9075"
	picture = HRE_eventPicture

	major = yes
	is_triggered_only = yes
	
	option = {
		name = "EVTOPTA9075"
		change_tag = HLR
		add_country_modifier = {
			name = "rome_reborn"
			duration = -1
		}
		every_country = {
			limit = {
				is_part_of_hre = yes
				is_emperor = no
				num_of_cities = 1
			}
			country_event = { id = 9076 }
		}
		every_province = {
			limit = {
				is_part_of_hre = yes
			}
			if = {
				limit = {
					owner = {
						is_subject_of = HLR
					}
				}
				cede_province = HLR
				add_core = HLR
				add_local_autonomy = -10
				if = { limit = { owner = { government = theocracy } } set_province_flag = cu_hlr_former_member_the }
				else_if = {  limit = { owner = { has_reform = free_city } } set_province_flag = cu_hlr_former_member_fre }
				else = { set_province_flag = cu_hlr_former_member_els }
			}
			else = { add_permanent_claim = HLR }
		}
		dismantle_hre = yes
		set_government_rank = 3
		add_adm_power = 250
		if = {
				limit = { has_custom_ideas = no }
				country_event = { id = ideagroups.1 } #Swap Ideas
			}
	}
}

#The Shadow Kingdom
country_event = {
	id = hre_event.1
	title = "hre_event.1.t"
	desc = "hre_event.1.d"
	picture = HRE_eventPicture

	trigger = {
		NOT = { has_global_flag = italy_out_of_HRE }
		normal_or_historical_nations = yes
		is_year = 1490
		NOT = { is_year = 1550 }
		capital_scope = {
			region = italy_region
		}
		#Adding these areas entirely into the Empire prevents this event.
		OR = {
			tuscany_area = {	
				is_part_of_hre = no
			}
			emilia_romagna_area = {	
				is_part_of_hre = no
			}
			liguria_area = {	
				is_part_of_hre = no
			}
			piedmont_area = {	
				is_part_of_hre = no
			}
			lombardy_area = {	
				is_part_of_hre = no
			}
			2977 = {
				is_part_of_hre = no
			}
			119 = {
				is_part_of_hre = no
			}
			2976 = {
				is_part_of_hre = no
			}
			108 = {	
				is_part_of_hre = no
			}
			1774 = {	
				is_part_of_hre = no
			}
			111 = {	
				is_part_of_hre = no
			}
			118 = {	#Rome
				is_part_of_hre = no
			}
		}
		is_part_of_hre = yes
		NOT = {	#Won't happen unless atleast one Italian HRE state isn't a vassal of a German HRE state.
			any_country = {
				overlord_of = ROOT
				is_part_of_hre = yes
				cu_germanic_group = yes
			}
		}
	}
	
	mean_time_to_happen = {
		months = 12
	}

	immediate = {
		hidden_effect = {
			set_global_flag = italy_out_of_HRE
			set_country_flag = shadow_kingdom #To keep track of info events
			every_country = {
				limit = {
					is_part_of_hre = yes
					OR = {
						culture_group = latin
						capital_scope = {
							region = italy_region
						}
					}
					NOT = { has_country_flag = shadow_kingdom }
				}
				country_event = { id = hre_event.2 }
				set_country_flag = shadow_kingdom #To keep track of info events
			}
			every_country = {
				limit = {
					OR = {
						culture_group = latin
						capital_scope = {
							region = italy_region
						}
					}
					is_part_of_hre = no
					any_owned_province = {
						is_part_of_hre = yes
					}
				}
				country_event = { id = hre_event.3 } #Clears Italian provinces for countries not in the HRE
				# Keep an eye on this, triggering country events instantly in immediate can cause loops
			}
		}
	}

	option = {			# 
		name = "hre_event.1.a"
		ai_chance = { factor = 1 }
		set_country_flag = left_the_empire #To keep track of opinions
		set_in_empire = no
		every_country = {
			limit = {
				ai = no
				overlord_of = ROOT
				NOT = { has_country_flag = shadow_kingdom }
				NOT = { has_country_flag = had_shadow_kingdom_info }
			}
			country_event = { id = hre_event.3 } #Info Event
		}
		every_neighbor_country = {
			limit = {
				ai = no
				NOT = { overlord_of = ROOT }
				NOT = { has_country_flag = shadow_kingdom }
				NOT = { has_country_flag = had_shadow_kingdom_info }
			}
			country_event = { id = hre_event.3 } #Info Event
		}
		if = {
			limit = {
				is_elector = yes
			}
			elector = no
		}
		emperor = {
			add_opinion = { who = ROOT modifier = slipped_out_of_empire }
		}
		add_opinion = { who = emperor modifier = slipped_out_of_empire }
		add_prestige = 25
	}
	option = {			#Our country is a natural part of the Empire
		name = "hre_event.1.b"
		ai_chance = { factor = 0 }
		custom_tooltip = PENALTY_SHADOW_KINGDOM
		add_opinion = {
			who = emperor
			modifier = stayed_in_the_empire
		}
		emperor = {
			add_imperial_influence = 5
			add_opinion = {
				who = ROOT
				modifier = stayed_in_the_empire
			}
		}
		every_country = {
			limit = {
				has_country_flag = left_the_empire
			}
			add_opinion = {
				who = ROOT
				modifier = stayed_in_the_empire_hostile
			}
		}
	}
}
