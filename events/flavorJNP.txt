#The Kalpi War
country_event = {
	id = flavor_jnp.8
	title = "flavor_jnp.8.t"
	desc = "flavor_jnp.8.d"
	picture = ACCUSATION_eventPicture
	
	fire_only_once = yes

	trigger = {
		tag = JNP
		NOT = { is_year = 1455 }
		is_subject = no
		religion_group = muslim
		NOT = { owns = 2059 }
		exists = MLW
		MLW = {
			is_subject = no
			religion_group = muslim
			NOT = { alliance_with = JNP }
		}
		BND = {
			vassal_of = MLW
			is_neighbor_of = JNP
		}
	}
	
	mean_time_to_happen = {
		months = 33
	}
	
	option = {
		name = "flavor_jnp.8.a"	#Send Ultimatum to Malwa
		ai_chance = { 
			factor = 50
			modifier = {
				factor = 0
				is_at_war = yes
			}
		}
		MLW = {
			country_event = { id = flavor_jnp.9 days = 14 }
		}
	}
	
	option = {
		name = "flavor_jnp.8.b"	#Let the matter rest.
		ai_chance = {
			factor = 50
		}
		add_prestige = -5
	}
}
